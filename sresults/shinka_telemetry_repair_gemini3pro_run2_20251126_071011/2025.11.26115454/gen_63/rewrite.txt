"""
Network telemetry repair algorithm using Dual-Flow Consensus with Quality Arbitration.
"""
from typing import Dict, Any, Tuple, List
import math

def repair_network_telemetry(telemetry: Dict[str, Dict[str, Any]], 
                             topology: Dict[str, List[str]]) -> Dict[str, Dict[str, Tuple]]:
    """
    Repair network telemetry using Dual-Flow Consensus and Quality-Weighted Arbitration.
    
    Strategy:
    1. Status Repair: Infer UP status from traffic; propagate DOWN status from Peer if Local is idle.
    2. Rate Repair (Iterative):
       - Calculate 'Flow Implied' targets for each interface based on Local Router balance.
       - Calculate 'Router Quality' for Local and Remote routers.
       - Arbitration Logic:
         - Enforce strict Physics Constraints (RX <= Peer TX, TX >= Peer RX).
         - If Consensus (Self ~= Peer): Average.
         - If Disagreement:
           - Check 'Flow Solidity': Does Peer value fix the Local flow imbalance?
           - Check 'Remote Quality': Is the Remote router strictly balanced?
           - If Local Flow and Remote Quality both support Peer -> Trust Peer (High Confidence).
           - If Local Flow supports Self -> Trust Self (Real Loss).
    3. Confidence Calibration:
       - Hybrid Score: Base category score (Verified, Supported, Ambiguous) minus Residual Flow Penalty.
    """
    
    # --- Configuration ---
    HARDENING_THRESHOLD = 0.02   # 2% relative error considered 'match'
    PHYSICS_THRESHOLD = 0.001    # 0.1% strict tolerance for physical impossibilities
    BASE_NOISE_FLOOR = 10.0      # Minimum Mbps to consider 'active'
    ITERATIONS = 4               # Convergence count
    
    # --- Helper: Dynamic Noise Floor ---
    def get_noise_floor(rate_a, rate_b=0.0):
        # Scale noise floor for high speed links, but keep base floor
        mx = max(rate_a, rate_b)
        return max(BASE_NOISE_FLOOR, mx * 0.001)

    # --- Helper: Normalized Error ---
    def calc_error(v1, v2):
        nf = get_noise_floor(v1, v2)
        return abs(v1 - v2) / max(v1, v2, nf)

    # --- Step 1: Initialization ---
    state = {}
    for if_id, data in telemetry.items():
        state[if_id] = {
            'rx': float(data.get('rx_rate', 0.0)),
            'tx': float(data.get('tx_rate', 0.0)),
            'status': data.get('interface_status', 'unknown'),
            'peer': data.get('connected_to'),
            'router': data.get('local_router'),
            'remote_router': data.get('remote_router'),
            'orig_rx': float(data.get('rx_rate', 0.0)),
            'orig_tx': float(data.get('tx_rate', 0.0)),
            'orig_status': data.get('interface_status', 'unknown')
        }

    # --- Step 2: Robust Status Repair ---
    for if_id, s in state.items():
        # Evidence
        local_traffic = s['rx'] > BASE_NOISE_FLOOR or s['tx'] > BASE_NOISE_FLOOR
        
        peer_traffic = False
        peer_is_down = False
        if s['peer'] and s['peer'] in state:
            p = state[s['peer']]
            if p['orig_rx'] > BASE_NOISE_FLOOR or p['orig_tx'] > BASE_NOISE_FLOOR:
                peer_traffic = True
            if p['orig_status'] == 'down':
                peer_is_down = True
        
        # Decision Matrix
        if local_traffic or peer_traffic:
            s['status'] = 'up'
        elif peer_is_down and not local_traffic:
            s['status'] = 'down'
        # Else: keep original
        
        # Consistency enforce
        if s['status'] != 'up':
            s['rx'] = 0.0
            s['tx'] = 0.0

    # --- Step 3: Iterative Rate Repair ---
    for _ in range(ITERATIONS):
        
        # 3.1: Calculate Router States (Imbalance & Quality)
        router_stats = {}
        for r_id, if_ids in topology.items():
            sum_rx = sum(state[i]['rx'] for i in if_ids if i in state)
            sum_tx = sum(state[i]['tx'] for i in if_ids if i in state)
            total_vol = max(sum_rx, sum_tx, BASE_NOISE_FLOOR)
            imbalance = sum_rx - sum_tx
            # Quality: 1.0 = Perfect. 0.0 = >5% Imbalance.
            # Using a steep curve to differentiate 'good' (locked) vs 'bad' (repairable) routers
            quality = max(0.0, 1.0 - (abs(imbalance) / total_vol * 20.0))
            router_stats[r_id] = {
                'imbalance': imbalance,
                'quality': quality
            }

        next_values = {}

        for if_id, s in state.items():
            if s['status'] != 'up':
                next_values[if_id] = {'rx': 0.0, 'tx': 0.0}
                continue

            peer_id = s['peer']
            r_id = s['router']
            rem_r_id = s['remote_router']
            has_peer = peer_id and peer_id in state
            
            # --- Gather Context ---
            # Local Flow Info
            local_target_rx = None
            local_target_tx = None
            local_q = 0.5
            if r_id in router_stats:
                rs = router_stats[r_id]
                local_q = rs['quality']
                # To balance: RX_new = RX_old - Imbalance
                local_target_rx = max(0.0, s['rx'] - rs['imbalance'])
                # To balance: TX_new = TX_old + Imbalance
                local_target_tx = max(0.0, s['tx'] + rs['imbalance'])
            
            # Remote Quality
            remote_q = 0.5
            if rem_r_id and rem_r_id in router_stats:
                remote_q = router_stats[rem_r_id]['quality']

            # --- RX Repair ---
            # Constraint: RX <= Peer TX
            val_self = s['rx']
            val_peer = state[peer_id]['tx'] if has_peer else None
            
            final_rx = val_self
            
            if val_peer is not None:
                # 1. Physics Violation (RX > Peer TX)
                if val_self > val_peer * (1.0 + PHYSICS_THRESHOLD):
                    final_rx = val_peer
                
                # 2. Agreement
                elif calc_error(val_self, val_peer) < HARDENING_THRESHOLD:
                    final_rx = (val_self + val_peer) / 2.0
                    
                # 3. Disagreement (RX < Peer TX) - Ambiguous
                else:
                    # Arbitration Candidates: Peer (Phantom Loss) vs Self (Real Loss)
                    if local_target_rx is not None:
                        err_peer_flow = calc_error(val_peer, local_target_rx)
                        err_self_flow = calc_error(val_self, local_target_rx)
                        
                        # Simplified decision based on dominant factor
                        if err_peer_flow < HARDENING_THRESHOLD:
                             # Peer fits my flow perfectly -> Trust Peer
                             final_rx = val_peer
                        elif err_self_flow < HARDENING_THRESHOLD:
                             # Self fits my flow perfectly -> Trust Self
                             final_rx = val_self
                        else:
                            # Neither fits perfectly. Who is better?
                            # If Remote is stable, trust Peer.
                            if remote_q > local_q and remote_q > 0.8:
                                final_rx = val_peer
                            elif err_peer_flow < err_self_flow:
                                final_rx = val_peer
                            else:
                                final_rx = val_self
                    else:
                        final_rx = val_peer

            # --- TX Repair ---
            # Constraint: TX >= Peer RX
            val_self = s['tx']
            val_peer = state[peer_id]['rx'] if has_peer else None
            
            final_tx = val_self
            
            if val_peer is not None:
                # 1. Physics Violation (TX < Peer RX)
                if val_self < val_peer * (1.0 - PHYSICS_THRESHOLD):
                    final_tx = val_peer
                    
                # 2. Agreement
                elif calc_error(val_self, val_peer) < HARDENING_THRESHOLD:
                    final_tx = (val_self + val_peer) / 2.0
                    
                # 3. Disagreement (TX > Peer RX)
                else:
                    if local_target_tx is not None:
                        err_peer_flow = calc_error(val_peer, local_target_tx)
                        err_self_flow = calc_error(val_self, local_target_tx)
                        
                        if err_peer_flow < HARDENING_THRESHOLD:
                            final_tx = val_peer
                        elif err_self_flow < HARDENING_THRESHOLD:
                            final_tx = val_self
                        else:
                            if remote_q > local_q and remote_q > 0.8:
                                final_tx = val_peer
                            elif err_peer_flow < err_self_flow:
                                final_tx = val_peer
                            else:
                                final_tx = val_self
                    else:
                        final_tx = val_peer

            next_values[if_id] = {'rx': final_rx, 'tx': final_tx}

        # Apply updates
        for if_id, vals in next_values.items():
            state[if_id]['rx'] = vals['rx']
            state[if_id]['tx'] = vals['tx']

    # --- Step 4: Confidence Calibration ---
    result = {}
    
    # Recalculate Final Flow Quality
    final_router_qual = {}
    for r_id, if_ids in topology.items():
        sum_rx = sum(state[i]['rx'] for i in if_ids if i in state)
        sum_tx = sum(state[i]['tx'] for i in if_ids if i in state)
        vol = max(sum_rx, sum_tx, BASE_NOISE_FLOOR)
        imb = abs(sum_rx - sum_tx) / vol
        final_router_qual[r_id] = max(0.0, 1.0 - (imb * 10.0))

    for if_id, s in state.items():
        orig_rx = s['orig_rx']
        orig_tx = s['orig_tx']
        
        peer_id = s['peer']
        has_peer = peer_id and peer_id in state
        
        peer_tx = state[peer_id]['tx'] if has_peer else None
        peer_rx = state[peer_id]['rx'] if has_peer else None
        
        r_id = s['router']
        local_q = final_router_qual.get(r_id, 0.5)
        remote_q = final_router_qual.get(s['remote_router'], 0.5) if s['remote_router'] else 0.5

        def get_confidence(final, orig, peer_val, l_q, r_q):
            # Error metric
            dist_orig = calc_error(final, orig)
            
            matches_peer = False
            if peer_val is not None:
                if calc_error(final, peer_val) < HARDENING_THRESHOLD:
                    matches_peer = True
            
            # Base Confidence
            conf = 1.0
            
            if dist_orig > HARDENING_THRESHOLD:
                # CASE: REPAIRED
                if matches_peer:
                    # Aligned with Peer. High confidence.
                    # Boost if routers are healthy.
                    conf = 0.90 + (0.05 * l_q) + (0.04 * r_q)
                else:
                    # Repaired to something else (e.g. Flow Target).
                    # Suspicious unless Local Flow is perfect.
                    if l_q > 0.9:
                        conf = 0.85
                    else:
                        conf = 0.60
            else:
                # CASE: KEPT ORIGINAL
                if peer_val is not None and not matches_peer:
                    # Disagreement.
                    # Only valid if Local Flow strongly supports it.
                    if l_q > 0.9:
                        # Verified by flow.
                        conf = 0.95
                    elif l_q > 0.7:
                        conf = 0.80
                    else:
                        conf = 0.65 # Ambiguous
                else:
                    # Agreement or No Peer
                    conf = 1.0
            
            # Hybrid Penalty: Cap score if local router is messy
            # (Prevent overconfidence in bad environments)
            if l_q < 0.5:
                conf = min(conf, 0.80)
            
            return max(0.0, min(1.0, conf))

        rx_conf = get_confidence(s['rx'], orig_rx, peer_tx, local_q, remote_q)
        tx_conf = get_confidence(s['tx'], orig_tx, peer_rx, local_q, remote_q)
        
        st_conf = 1.0
        if s['status'] != s['orig_status']:
            st_conf = 0.95

        result[if_id] = {
            'rx_rate': (orig_rx, s['rx'], rx_conf),
            'tx_rate': (orig_tx, s['tx'], tx_conf),
            'interface_status': (s['orig_status'], s['status'], st_conf),
            'connected_to': telemetry[if_id].get('connected_to'),
            'local_router': telemetry[if_id].get('local_router'),
            'remote_router': telemetry[if_id].get('remote_router')
        }

    return result
