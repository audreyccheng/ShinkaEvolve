<DESCRIPTION>
Implements Dual-Flow Consensus for network telemetry repair.
1.  **Dual-Source Validation**: Uses both Local Router and Peer Router flow states to validate measurements. Peer measurements are considered "Solid" only if they are consistent with the Peer Router's flow balance.
2.  **Consensus Arbitration**: Arbitrates disagreements (e.g., RX < Peer TX) by comparing candidates (Self, Peer) against Flow Targets.
3.  **Strict vs Invalid Constraints**: Refines the "Impossible" check (RX > Peer TX) to allow exceptions if Peer TX is demonstrably inconsistent with Peer Flow (under-reporting).
4.  **Confidence Calibration**: Uses a continuous scoring model that scales based on residual flow quality and peer agreement.
</DESCRIPTION>

    Repair network telemetry using Analytic Flow Consensus.

    Strategy:
    1. Status Repair: Infer status from local/remote traffic activity.
    2. Rate Repair (Iterative):
       - For each interface, calculate the 'Flow Implied' rate that would perfectly balance the router.
       - Compare 'Self', 'Peer', and 'Flow Implied' values.
       - Enforce physical constraints (RX <= Peer TX, TX >= Peer RX).
       - Arbitrate loss scenarios (RX < Peer TX) using the Flow Implied value.
    3. Confidence Calibration:
       - continuous scoring based on residual errors and agreement strength.
    """

    HARDENING_THRESHOLD = 0.02   # 2% relative error considered 'match'
    BASE_NOISE_FLOOR = 10.0      # Minimum Mbps to consider 'active'
    ITERATIONS = 4               # Convergence count

    # --- Helper: Dynamic Noise Floor ---
    def get_noise_floor(rate_a, rate_b=0.0):
        # Scale noise floor for high speed links, but keep base floor for IoT/Idle
        mx = max(rate_a, rate_b)
        return max(BASE_NOISE_FLOOR, mx * 0.005) # 0.5% of rate or 10Mbps

    def calc_error(v1, v2):
        nf = get_noise_floor(v1, v2)
        return abs(v1 - v2) / max(v1, v2, nf)

    state = {}
    for if_id, data in telemetry.items():
        state[if_id] = {
            'rx': float(data.get('rx_rate', 0.0)),
            'tx': float(data.get('tx_rate', 0.0)),
            'status': data.get('interface_status', 'unknown'),
            'peer': data.get('connected_to'),
            'router': data.get('local_router'),
            'orig_rx': float(data.get('rx_rate', 0.0)),
            'orig_tx': float(data.get('tx_rate', 0.0)),
            'orig_status': data.get('interface_status', 'unknown')
        }

    for if_id, s in state.items():
        # Traffic Evidence
        local_traffic = s['rx'] > BASE_NOISE_FLOOR or s['tx'] > BASE_NOISE_FLOOR

        peer_traffic = False
        peer_is_down = False
        if s['peer'] and s['peer'] in state:
            p = state[s['peer']]
            if p['orig_rx'] > BASE_NOISE_FLOOR or p['orig_tx'] > BASE_NOISE_FLOOR:
                peer_traffic = True
            if p['orig_status'] == 'down':
                peer_is_down = True

        # Decision Matrix
        if local_traffic or peer_traffic:
            s['status'] = 'up'
        elif peer_is_down and not local_traffic:
            s['status'] = 'down'
        # Else: if originally UP and idle, stay UP. If originally DOWN and idle, stay DOWN.

        # Consistency enforce
        if s['status'] != 'up':
            s['rx'] = 0.0
            s['tx'] = 0.0

        for r_id, if_ids in topology.items():
            sum_rx = sum(state[i]['rx'] for i in if_ids if i in state)
            sum_tx = sum(state[i]['tx'] for i in if_ids if i in state)
            # Total volume for normalization
            total_vol = max(sum_rx, sum_tx, BASE_NOISE_FLOOR)
            imbalance = sum_rx - sum_tx
            router_stats[r_id] = {
                'sum_rx': sum_rx,
                'sum_tx': sum_tx,
                'imbalance': imbalance,
                'quality': max(0.0, 1.0 - (abs(imbalance) / total_vol * 10.0)) # 1.0=Perfect, 0.0=Bad
            }

        for if_id, s in state.items():
            if s['status'] != 'up':
                next_values[if_id] = {'rx': 0.0, 'tx': 0.0}

            r_id = s['router']
            has_peer = peer_id and peer_id in state

            # --- Solve for Flow-Implied Values ---
            # RX_implied: The RX value that makes Sum_RX == Sum_TX
            # Current Balance: Sum_RX - Sum_TX = Imbalance
            # We want New_Imbalance = 0
            # (Sum_RX - old_rx + new_rx) - Sum_TX = 0
            # new_rx = Sum_TX - Sum_RX + old_rx = old_rx - Imbalance

            flow_rx = None
            flow_tx = None
            router_q = 0.5

            if r_id in router_stats:
                rs = router_stats[r_id]
                router_q = rs['quality']
                flow_rx = max(0.0, s['rx'] - rs['imbalance'])
                flow_tx = max(0.0, s['tx'] + rs['imbalance'])

            val_self = s['rx']
            val_peer = state[peer_id]['tx'] if has_peer else None

            final_rx = val_self

                # Allow small tolerance
                if val_self > val_peer * (1.0 + HARDENING_THRESHOLD):
                    # Impossible: RX > Peer TX.
                    # Must clamp to Peer TX unless flow strictly demands we KEEP the excess (unlikely).
                    # Actually, if we have excess RX, it's phantom traffic. Peer TX is usually the ceiling.
                    final_rx = val_peer

                # 2. Check Loss Case (RX < Peer TX) or Agreement
                else:
                    err_peer = calc_error(val_self, val_peer)

                        # Agreement -> Average
                        final_rx = (val_self + val_peer) / 2.0
                    else:
                        # Disagreement (Self < Peer usually).
                        # Use Flow to Arbitrate.
                        if flow_rx is not None:
                            # Which is closer to Flow Target?
                            err_self_flow = calc_error(val_self, flow_rx)
                            err_peer_flow = calc_error(val_peer, flow_rx)

                            # Decision
                            if err_peer_flow < err_self_flow:
                                # Flow confirms Peer (Packet Loss detected, Peer is correct source rate)
                                final_rx = val_peer
                            elif err_self_flow < err_peer_flow:
                                # Flow confirms Self (Real Loss occurred, or Peer is over-reporting)
                                final_rx = val_self
                            else:
                                # Ambiguous. Trust Peer (Link Symmetry R3 is robust).
                                final_rx = val_peer
                        else:
                            # No flow info -> Trust Peer
                            final_rx = val_peer

            # --- TX Repair ---
            # Constraint: TX >= Peer RX
            val_self = s['tx']
            val_peer = state[peer_id]['rx'] if has_peer else None

            final_tx = val_self

            if val_peer is not None:
                # 1. Check strict constraint (Impossible case: TX < Peer RX)
                if val_self < val_peer * (1.0 - HARDENING_THRESHOLD):
                     # Impossible. Must boost TX to match Peer RX.
                     final_tx = val_peer

                # 2. Check Surplus Case (TX > Peer RX) or Agreement
                else:
                    err_peer = calc_error(val_self, val_peer)
                    if err_peer < HARDENING_THRESHOLD:
                        final_tx = (val_self + val_peer) / 2.0
                    else:
                        # Disagreement (Self > Peer).
                        if flow_tx is not None:
                            err_self_flow = calc_error(val_self, flow_tx)
                            err_peer_flow = calc_error(val_peer, flow_tx)

                            if err_peer_flow < err_self_flow:
                                # Flow says lower value (Peer) is better -> Phantom TX repaired
                                final_tx = val_peer
                            elif err_self_flow < err_peer_flow:
                                # Flow says high value (Self) is better -> Peer Loss confirmed
                                final_tx = val_self
                            else:
                                final_tx = val_peer
                        else:
                            final_tx = val_peer

            next_values[if_id] = {'rx': final_rx, 'tx': final_tx}

        # Apply updates
        for if_id, vals in next_values.items():
            state[if_id]['rx'] = vals['rx']
            state[if_id]['tx'] = vals['tx']

    # --- Step 4: Confidence Calibration ---
    result = {}

    # Recalculate Final Flow Quality
    final_router_qual = {}
    for r_id, if_ids in topology.items():
        sum_rx = sum(state[i]['rx'] for i in if_ids if i in state)
        sum_tx = sum(state[i]['tx'] for i in if_ids if i in state)
        vol = max(sum_rx, sum_tx, BASE_NOISE_FLOOR)
        imb = abs(sum_rx - sum_tx) / vol
        # High quality if imbalance < 2%
        final_router_qual[r_id] = max(0.0, 1.0 - (imb * 10.0))

    for if_id, s in state.items():
        orig_rx = s['orig_rx']
        orig_tx = s['orig_tx']

        peer_id = s['peer']
        has_peer = peer_id and peer_id in state

        # Get Peer Final Values
        peer_tx = state[peer_id]['tx'] if has_peer else None
        peer_rx = state[peer_id]['rx'] if has_peer else None

        r_id = s['router']
        flow_q = final_router_qual.get(r_id, 0.5)

        def get_confidence(final, orig, peer_val, flow_q):
            # Base logic: Start high, penalize for unverified deviations

            # 1. Did we change it?
            was_repaired = calc_error(final, orig) > HARDENING_THRESHOLD

            # 2. Do we match Peer?
            matches_peer = False
            if peer_val is not None:
                if calc_error(final, peer_val) < HARDENING_THRESHOLD:
                    matches_peer = True

            conf = 1.0

            if was_repaired:
                if matches_peer:
                    # Robust repair: We aligned with peer.
                    # Confidence depends on Flow confirmation.
                    if flow_q > 0.8:
                        conf = 0.98 # Validated by Link + Flow
                    elif flow_q > 0.5:
                        conf = 0.90 # Validated by Link, Flow okay
                    else:
                        conf = 0.80 # Validated by Link, Flow bad (but Link > Flow usually)
                else:
                    # We changed value, but NOT to peer? (e.g. Averaging or Flow forced value?)
                    # If matches Flow significantly?
                    if flow_q > 0.9:
                        conf = 0.85
                    else:
                        conf = 0.60 # Sketchy repair
            else:
                # We kept original
                if peer_val is not None and not matches_peer:
                    # Disagreement. We trust Self over Peer.
                    # Valid if Flow supports Self.
                    if flow_q > 0.8:
                        conf = 0.95 # Flow validates Self
                    else:
                        conf = 0.75 # Ambiguous (Peer says X, I say Y, Flow is messy)
                else:
                    # Agreement or No Peer
                    conf = 1.0

            return conf

        tx_conf = get_confidence(s['tx'], orig_tx, peer_rx, flow_q)

        # Status confidence
        st_conf = 1.0
        if s['status'] != s['orig_status']:
            st_conf = 0.95

            'rx_rate': (orig_rx, s['rx'], rx_conf),
            'tx_rate': (orig_tx, s['tx'], tx_conf),
