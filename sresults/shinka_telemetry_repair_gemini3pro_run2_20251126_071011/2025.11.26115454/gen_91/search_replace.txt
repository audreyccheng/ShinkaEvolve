<DESCRIPTION>
1. Implements Symmetric Flow Consensus: Explicitly calculates flow targets for both Local and Remote routers to identify "Golden Truth" values that satisfy both ends.
2. Differential Quality Override: Trusts the Peer value if the Remote router is significantly healthier than the Local router.
3. Post-Repair Residual Damping: Penalizes confidence scores if the local router remains unbalanced after repair.

        def get_confidence(final, orig, peer_val, l_q, r_q):
            # Error metric
            dist_orig = calc_error(final, orig)

            # Peer match?
            matches_peer = False
            if peer_val is not None and calc_error(final, peer_val) < HARDENING_THRESHOLD:
                matches_peer = True

            # Base Confidence Logic
            conf = 1.0

            if dist_orig > HARDENING_THRESHOLD:
                # REPAIRED
                if matches_peer:
                    # Aligned with peer.
                    # Base: 0.88 + Local Q + Remote Q
                    conf = 0.88 + (0.07 * l_q) + (0.04 * r_q)
                else:
                    # Repaired but NOT matching peer.
                    if l_q > 0.9: conf = 0.85
                    else: conf = 0.60
            else:
                # KEPT ORIGINAL
                if peer_val is not None and not matches_peer:
                    # Disagreement. Trusted Self.
                    if l_q > 0.9:
                        if r_q > 0.8: conf = 0.88 # Stalemate
                        else: conf = 0.95 # I'm right
                    elif l_q > 0.7:
                        conf = 0.75
                    else:
                        conf = 0.60 # Ambiguous
                else:
                    # Agreement or No Peer
                    conf = 1.0

            # Post-Repair Damping: Penalize if local router remains unbalanced
            # If our confidence relied on "Flow Arbitration" but the flow didn't end up balanced,
            # we should reduce confidence.
            if l_q < 0.8:
                conf *= (0.8 + 0.2 * l_q)

            return max(0.0, min(1.0, conf))
