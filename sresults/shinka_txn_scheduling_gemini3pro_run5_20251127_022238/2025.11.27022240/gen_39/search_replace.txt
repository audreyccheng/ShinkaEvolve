<DESCRIPTION>
Optimizes the scheduling algorithm by integrating several high-performance strategies:
1.  **Constructive Phase**: Implements 'Adaptive Gamma Decay' (decaying from 1.5 to 1.0) to transition from work-density packing to best-fit filling. Adds a 'Zero-Cost Bonus' to heavily prioritize transactions that fit into parallel slots without increasing makespan.
2.  **Refinement Phase**: Replaces the single-block Ruin and SA with a **Multi-Segment Ruin** (removing two disjoint blocks to allow teleporting) and a **Stochastic Variable Neighborhood Descent (VND)**. The VND cycles through Shift, Swap, and Reverse neighborhoods, restarting on improvement to aggressively exploit local optima.

def get_best_schedule(workload, num_seqs):

    Hybrid Scheduling Algorithm:
    1. Constructive Phase: Wide Beam Search with diversity pruning to build a robust initial schedule.
    2. Refinement Phase: Simulated Annealing with advanced operators (Block Moves) to optimize clusters.

    Args:
        workload: Workload object containing transaction data
        num_seqs: Hint for computational budget

    Returns:
        Tuple of (lowest makespan, corresponding schedule)
    """

    num_txns = workload.num_txns

    # Extract transaction costs
    txn_lengths = {}
    for i in range(num_txns):
        try:
            txn_lengths[i] = workload.txns[i][0][3]
        except (IndexError, TypeError, AttributeError):
            txn_lengths[i] = 1.0

    BEAM_WIDTH = max(16, int(num_seqs * 2))
    SAMPLE_SIZE = 16
    GAMMA = 1.3

    seeds = set(lpt_sorted_indices[:BEAM_WIDTH])
    if len(seeds) < BEAM_WIDTH:
        remaining_slots = BEAM_WIDTH - len(seeds)
        seeds.update(random.sample(range(num_txns), min(num_txns - len(seeds), remaining_slots)))

    beam = []
    for t in seeds:
        seq = [t]
        cost = workload.get_opt_seq_cost(seq)
        acc_work = txn_lengths[t]
        score = cost - (GAMMA * acc_work)

        remaining = set(range(num_txns))
        remaining.remove(t)

        beam.append({
            'seq': seq,
            'cost': cost,
            'score': score,
            'acc_work': acc_work,
            'rem': remaining
        })

    beam = beam[:BEAM_WIDTH]

                    added += 1

            base_work = parent['acc_work']

                new_rem.remove(t)

                    'seq': new_seq,
                    'cost': new_cost,
                    'score': new_score,
                    'acc_work': new_work,
                    'rem': new_rem
                })

            break

        candidates.sort(key=lambda x: x['score'])

        k_best = int(BEAM_WIDTH * 0.7)
        next_beam = candidates[:k_best]

        remaining_needed = BEAM_WIDTH - len(next_beam)
        if remaining_needed > 0:
            # Sample from a wider pool of good candidates to maintain diversity
            pool_size = min(len(candidates), BEAM_WIDTH * 3)
            pool = candidates[k_best : pool_size]
            if len(pool) <= remaining_needed:
                next_beam.extend(pool)
            else:
                next_beam.extend(random.sample(pool, remaining_needed))

    current_cost = best_candidate['cost']

    # Replaces simple SA with a Ruin-and-Recreate strategy + Local Search

    # Configuration
    ILS_CYCLES = 5  # Number of ruin-recreate loops
    SA_STEPS = 120  # Steps of local search after recreation

    best_schedule = list(current_schedule)
    best_cost = current_cost

    # We maintain current_schedule as the working solution

    for cycle in range(ILS_CYCLES):
        # A. Perturbation: Ruin and Recreate (Skip first cycle to purely optimize beam result)
        if cycle > 0:
            # Ruin: Remove a contiguous block
            # Size: 10% to 20% of transactions
            block_size = random.randint(max(2, int(num_txns * 0.1)), max(3, int(num_txns * 0.2)))
            start_idx = random.randint(0, num_txns - block_size)

            # Extract transactions to re-insert
            removed_txns = current_schedule[start_idx : start_idx + block_size]
            del current_schedule[start_idx : start_idx + block_size]

            # Shuffle to avoid bias
            random.shuffle(removed_txns)

            # Recreate: Greedy Best-Fit Insertion
            # For each txn, find best position in current partial schedule
            for txn in removed_txns:
                best_pos = -1
                best_incr_cost = float('inf')

                # Try all valid positions
                # Optimization: We know removing a txn generally reduces/keeps cost,
                # adding it increases. We want min increase.
                for pos in range(len(current_schedule) + 1):
                    current_schedule.insert(pos, txn)
                    cost = workload.get_opt_seq_cost(current_schedule)
                    if cost < best_incr_cost:
                        best_incr_cost = cost
                        best_pos = pos
                    del current_schedule[pos]

                # Commit best insert
                current_schedule.insert(best_pos, txn)

            current_cost = workload.get_opt_seq_cost(current_schedule)

            # Update best if this greedy step found a better global solution (unlikely but possible)
            if current_cost < best_cost:
                best_cost = current_cost
                best_schedule = list(current_schedule)

        # B. Local Search: Simulated Annealing
        # Short burst of SA to fix local suboptimal orderings after insertion
        temp = max(1.0, current_cost * 0.03)
        cooling = 0.96

        for _ in range(SA_STEPS):
            neighbor = list(current_schedule)
            r = random.random()

            # Operators
            if r < 0.3: # Swap
                i, j = random.sample(range(num_txns), 2)
                neighbor[i], neighbor[j] = neighbor[j], neighbor[i]
            elif r < 0.7: # Insert (Shift)
                i = random.randint(0, num_txns - 1)
                val = neighbor.pop(i)
                j = random.randint(0, num_txns - 1)
                neighbor.insert(j, val)
            else: # Block Reorder (Reverse small chunk)
                bs = random.randint(2, 6)
                if num_txns > bs:
                    si = random.randint(0, num_txns - bs)
                    neighbor[si:si+bs] = reversed(neighbor[si:si+bs])

            # Evaluation
            n_cost = workload.get_opt_seq_cost(neighbor)
            delta = n_cost - current_cost

            accept = False
            if delta < 0:
                accept = True
            elif temp > 0.001 and random.random() < math.exp(-delta / temp):
                accept = True

            if accept:
                current_schedule = neighbor
                current_cost = n_cost
                if current_cost < best_cost:
                    best_cost = current_cost
                    best_schedule = list(current_schedule)

            temp *= cooling

        # C. Acceptance / Restart Strategy for next cycle
        # If we drifted too far from best, restart from best with some probability
        # This intensifies search around the global optimum
        if current_cost > best_cost:
            # 60% chance to revert to best, 40% chance to explore from current
            if random.random() < 0.6:
                current_schedule = list(best_schedule)
                current_cost = best_cost

    return best_cost, best_schedule

def get_best_schedule(workload, num_seqs):
    """
    Hybrid Scheduling Algorithm:
    1. Constructive Phase: Adaptive Work-Density Beam Search with Parallelism Bonus.
    2. Refinement Phase: Multi-Segment Ruin & Recreate with Stochastic VND.

    Args:
        workload: Workload object containing transaction data
        num_seqs: Hint for computational budget

    Returns:
        Tuple of (lowest makespan, corresponding schedule)
    """

    # --- 0. Helper & Precomputation ---
    num_txns = workload.num_txns

    # Extract transaction costs
    txn_lengths = {}
    for i in range(num_txns):
        try:
            txn_lengths[i] = workload.txns[i][0][3]
        except (IndexError, TypeError, AttributeError):
            txn_lengths[i] = 1.0

    lpt_sorted_indices = sorted(txn_lengths.keys(), key=lambda k: txn_lengths[k], reverse=True)

    # --- 1. Constructive Phase: Adaptive Beam Search ---

    # Beam parameters
    BEAM_WIDTH = max(16, int(num_seqs * 2.2))
    SAMPLE_SIZE = 12

    # Initial Beam Seeding
    seeds = set(lpt_sorted_indices[:BEAM_WIDTH])
    if len(seeds) < BEAM_WIDTH:
        remaining_slots = BEAM_WIDTH - len(seeds)
        seeds.update(random.sample(range(num_txns), min(num_txns - len(seeds), remaining_slots)))

    beam = []
    initial_gamma = 1.5

    for t in seeds:
        seq = [t]
        cost = workload.get_opt_seq_cost(seq)
        acc_work = txn_lengths[t]
        score = cost - (initial_gamma * acc_work)

        remaining = set(range(num_txns))
        remaining.remove(t)

        beam.append({
            'seq': seq,
            'cost': cost,
            'score': score,
            'acc_work': acc_work,
            'rem': remaining
        })

    beam.sort(key=lambda x: x['score'])
    beam = beam[:BEAM_WIDTH]

    # Construction Loop
    for step in range(num_txns - 1):
        # Adaptive Gamma: Decay from 1.5 to 1.0
        # High Gamma early emphasizes packing 'work' (LPT style)
        # Low Gamma late emphasizes 'fit' (minimizing cost increase)
        progress = step / num_txns
        current_gamma = 1.5 - (0.5 * progress)

        candidates = []

        for parent in beam:
            rem_set = parent['rem']
            if not rem_set:
                continue

            to_eval = set()

            # A) Heuristic candidates (Top LPT)
            added = 0
            for t in lpt_sorted_indices:
                if t in rem_set:
                    to_eval.add(t)
                    added += 1
                    if added >= 5: break

            # B) Random candidates (Diversity)
            rem_list = list(rem_set)
            needed = SAMPLE_SIZE - len(to_eval)
            if needed > 0 and rem_list:
                pool = [x for x in rem_list if x not in to_eval]
                count = min(len(pool), needed)
                to_eval.update(random.sample(pool, count))

            base_seq = parent['seq']
            base_work = parent['acc_work']
            parent_cost = parent['cost']

            for t in to_eval:
                new_seq = base_seq + [t]
                new_cost = workload.get_opt_seq_cost(new_seq)
                new_work = base_work + txn_lengths[t]

                # Base Score
                new_score = new_cost - (current_gamma * new_work)

                # Zero-Cost Bonus (Parallelism)
                # If adding the txn increases cost by 0 (or negligible amount),
                # it means it was perfectly parallelized. Reward this heavily.
                if new_cost <= parent_cost + 1e-5:
                    # Bonus proportional to length of task successfully hidden
                    new_score -= (txn_lengths[t] * 2.5)

                new_rem = rem_set.copy()
                new_rem.remove(t)

                candidates.append({
                    'seq': new_seq,
                    'cost': new_cost,
                    'score': new_score,
                    'acc_work': new_work,
                    'rem': new_rem
                })

        if not candidates:
            break

        # Diversity-Aware Pruning
        candidates.sort(key=lambda x: x['score'])

        k_best = int(BEAM_WIDTH * 0.6)
        next_beam = candidates[:k_best]

        # Fill remainder with sampled candidates from the top tier (avoiding identical tails if possible)
        remaining_needed = BEAM_WIDTH - len(next_beam)
        if remaining_needed > 0:
            pool = candidates[k_best : min(len(candidates), BEAM_WIDTH * 4)]
            if len(pool) <= remaining_needed:
                next_beam.extend(pool)
            else:
                next_beam.extend(random.sample(pool, remaining_needed))

        beam = next_beam

    # Select best from construction
    beam.sort(key=lambda x: x['cost'])
    best_candidate = beam[0]

    current_schedule = best_candidate['seq']
    current_cost = best_candidate['cost']

    # --- 2. Refinement Phase: Multi-Segment ILS + Stochastic VND ---

    best_schedule = list(current_schedule)
    best_cost = current_cost

    # Adjust cycles based on workload size
    ILS_CYCLES = 5
    if num_txns < 20: ILS_CYCLES = 2

    for cycle in range(ILS_CYCLES):

        # A. Multi-Segment Ruin
        # Instead of removing one big block, remove 2 smaller disjoint blocks
        # This allows moving transactions across the schedule more effectively.
        work_seq = list(current_schedule)
        removed_txns = []

        # Determine number of blocks to remove
        num_blocks = 2 if num_txns > 25 else 1

        for _ in range(num_blocks):
            if len(work_seq) < 2: break
            # Size: 5% - 15% per block
            bs = max(1, int(len(best_schedule) * random.uniform(0.05, 0.15)))
            if len(work_seq) > bs:
                start = random.randint(0, len(work_seq) - bs)
                removed_txns.extend(work_seq[start : start + bs])
                del work_seq[start : start + bs]

        random.shuffle(removed_txns)

        # B. Recreate: Greedy Best-Fit
        for txn in removed_txns:
            best_pos = -1
            best_incr = float('inf')

            # Use strided search for very large schedules to save time
            step = 2 if len(work_seq) > 60 else 1
            positions = list(range(0, len(work_seq) + 1, step))
            if len(work_seq) not in positions: positions.append(len(work_seq))

            for pos in positions:
                work_seq.insert(pos, txn)
                c = workload.get_opt_seq_cost(work_seq)
                if c < best_incr:
                    best_incr = c
                    best_pos = pos
                del work_seq[pos]

            work_seq.insert(best_pos, txn)

        current_schedule = work_seq
        current_cost = best_incr # Approximate cost

        # Exact cost check
        real_cost = workload.get_opt_seq_cost(current_schedule)
        current_cost = real_cost

        if current_cost < best_cost:
            best_cost = current_cost
            best_schedule = list(current_schedule)

        # C. Stochastic Variable Neighborhood Descent (VND)
        # Cycle through neighborhoods: Shift -> Swap -> Reverse
        # Restart cycle if improvement found.

        neighborhoods = ['shift', 'swap', 'reverse']
        max_vnd_steps = 150 # Budget for this ILS cycle
        steps_taken = 0

        while steps_taken < max_vnd_steps:
            improved = False

            for nh in neighborhoods:
                # Try a batch of moves in this neighborhood
                batch_size = 20
                best_neighbor = None
                best_neighbor_cost = current_cost

                for _ in range(batch_size):
                    cand = list(current_schedule)

                    if nh == 'shift':
                        i = random.randint(0, len(cand)-1)
                        val = cand.pop(i)
                        j = random.randint(0, len(cand))
                        cand.insert(j, val)
                    elif nh == 'swap':
                        i, j = random.sample(range(len(cand)), 2)
                        cand[i], cand[j] = cand[j], cand[i]
                    elif nh == 'reverse':
                        if len(cand) < 4: continue
                        bs = random.randint(3, 8)
                        si = random.randint(0, len(cand) - bs)
                        cand[si:si+bs] = reversed(cand[si:si+bs])

                    c = workload.get_opt_seq_cost(cand)
                    steps_taken += 1

                    if c < best_neighbor_cost:
                        best_neighbor_cost = c
                        best_neighbor = cand

                # If we found an improvement in this batch, take it and restart VND cycle
                if best_neighbor and best_neighbor_cost < current_cost:
                    current_schedule = best_neighbor
                    current_cost = best_neighbor_cost
                    if current_cost < best_cost:
                        best_cost = current_cost
                        best_schedule = list(current_schedule)
                    improved = True
                    break # Restart neighborhood cycle

                if steps_taken >= max_vnd_steps: break

            if not improved:
                break # Local optimum reached for stochastic sampling

        # Restart Strategy: occasionally revert to global best
        if current_cost > best_cost and random.random() < 0.4:
            current_schedule = list(best_schedule)
            current_cost = best_cost

    return best_cost, best_schedule
