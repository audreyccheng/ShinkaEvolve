def get_best_schedule(workload, num_seqs):
    """
    Get optimal schedule using a Beam Search algorithm.

    Beam Search maintains a set of K partial schedules (the beam) and extends them
    in parallel, keeping only the best K extensions at each step. This allows
    exploring multiple promising paths simultaneously and backing out of local
    optima that a pure greedy approach would get stuck in.

    Args:
        workload: Workload object containing transaction data
        num_seqs: Used to derive beam width and search budget

    Returns:
        Tuple of (lowest makespan, corresponding schedule)
    """
    num_txns = workload.num_txns

    # 1. Heuristic Setup: Calculate transaction costs for LPT (Longest Processing Time)
    # We prioritize checking "heavy" transactions early in the candidate selection
    try:
        # Access internal structure: txn[0] is info, info[3] is length/cost
        txn_costs = [(i, workload.txns[i][0][3]) for i in range(num_txns)]
    except (IndexError, TypeError, AttributeError):
        # Fallback cost if structure varies
        txn_costs = [(i, 1) for i in range(num_txns)]

    # Sort by cost descending
    txn_costs.sort(key=lambda x: x[1], reverse=True)
    sorted_txn_indices = [x[0] for x in txn_costs]

    # 2. Search Parameters
    # Adjust BEAM_WIDTH based on the 'num_seqs' hint.
    # A width of 5-8 is usually efficient for this problem size (~100 items).
    # We balance Width vs Samples to keep execution time reasonable.
    BEAM_WIDTH = max(4, int(num_seqs * 0.8))
    SAMPLES_PER_NODE = 12
    HEURISTIC_COUNT = 4  # How many top-longest txns to force check

    # 3. Initialization
    # Start with a set of diverse single-transaction schedules
    # We pick distinct random starts to ensure the beam covers different parts of the space
    start_candidates = random.sample(range(num_txns), min(num_txns, BEAM_WIDTH * 2))

    beam = []
    for start_node in start_candidates:
        seq = [start_node]
        # Initial cost is trivial but good to have baseline
        cost = workload.get_opt_seq_cost(seq)
        remaining = set(range(num_txns))
        remaining.remove(start_node)
        beam.append({
            'cost': cost,
            'seq': seq,
            'rem': remaining
        })

    # Filter to top BEAM_WIDTH starts
    beam.sort(key=lambda x: x['cost'])
    beam = beam[:BEAM_WIDTH]

    # 4. Beam Search Loop
    # We construct the schedule step by step. Total steps = num_txns - 1
    for step in range(num_txns - 1):
        candidates_pool = []

        # Expand each partial schedule in the current beam
        for path in beam:
            current_seq = path['seq']
            remaining = path['rem']

            if not remaining:
                continue

            # Selection Strategy: Mix of Heuristic (LPT) and Random
            next_candidates = set()

            # A) Add Top Longest Remaining Transactions
            added_heuristic = 0
            for t in sorted_txn_indices:
                if t in remaining:
                    next_candidates.add(t)
                    added_heuristic += 1
                    if added_heuristic >= HEURISTIC_COUNT:
                        break

            # B) Fill rest with Random Sampling
            needed = SAMPLES_PER_NODE - len(next_candidates)
            rem_list = list(remaining)

            if needed > 0 and rem_list:
                # Filter out ones we already picked
                pool = [x for x in rem_list if x not in next_candidates]
                if len(pool) <= needed:
                    next_candidates.update(pool)
                else:
                    next_candidates.update(random.sample(pool, needed))

            # Evaluate all selected candidates
            for next_txn in next_candidates:
                # Create new extended sequence
                new_seq = current_seq + [next_txn]

                # Get Makespan (Cost) from Simulator
                # This accounts for all dependencies and conflicts
                new_cost = workload.get_opt_seq_cost(new_seq)

                new_rem = remaining.copy()
                new_rem.remove(next_txn)

                candidates_pool.append({
                    'cost': new_cost,
                    'seq': new_seq,
                    'rem': new_rem
                })

        # Pruning Phase
        if not candidates_pool:
            break

        # Sort all extensions by cost (lowest makespan is better)
        candidates_pool.sort(key=lambda x: x['cost'])

        # Select best K to form the new beam
        # We take the top BEAM_WIDTH sequences
        beam = candidates_pool[:BEAM_WIDTH]

    # 5. Final Result
    # Return the best schedule found
    best_result = beam[0]
    return best_result['cost'], best_result['seq']

    Get optimal schedule using an improved Beam Search algorithm.
    Includes diversity preservation and LPT-biased heuristics.

    Args:
        workload: Workload object containing transaction data
        num_seqs: Used to derive beam width and search budget

    Returns:
        Tuple of (lowest makespan, corresponding schedule)
    """
    num_txns = workload.num_txns

    # 1. Heuristic Setup
    try:
        # (index, length)
        txn_costs = [(i, workload.txns[i][0][3]) for i in range(num_txns)]
    except (IndexError, TypeError, AttributeError):
        txn_costs = [(i, 1) for i in range(num_txns)]

    cost_map = dict(txn_costs)

    # Sort by length descending (LPT)
    txn_costs.sort(key=lambda x: x[1], reverse=True)
    sorted_txn_indices = [x[0] for x in txn_costs]

    # 2. Search Parameters
    # Increased parameters for better exploration
    BEAM_WIDTH = max(8, int(num_seqs))
    SAMPLES_PER_NODE = 16
    HEURISTIC_COUNT = 6  # Increase heuristic bias

    # 3. Initialization
    # Initialize with top LPT transactions + some random ones to seed diversity
    start_pool = set(sorted_txn_indices[:BEAM_WIDTH])
    start_pool.update(random.sample(range(num_txns), min(num_txns, BEAM_WIDTH)))

    beam = []
    for start_node in start_pool:
        seq = [start_node]
        cost = workload.get_opt_seq_cost(seq)
        remaining = set(range(num_txns))
        remaining.remove(start_node)
        beam.append({
            'cost': cost,
            'seq': seq,
            'rem': remaining
        })

    # Keep best unique starts
    beam.sort(key=lambda x: x['cost'])
    beam = beam[:BEAM_WIDTH]

    # 4. Beam Search Loop
    for step in range(num_txns - 1):
        candidates_pool = []

        for path in beam:
            current_seq = path['seq']
            remaining = path['rem']

            if not remaining:
                continue

            # Candidate Generation
            next_candidates = set()

            # A) Top Longest Remaining
            added_heuristic = 0
            for t in sorted_txn_indices:
                if t in remaining:
                    next_candidates.add(t)
                    added_heuristic += 1
                    if added_heuristic >= HEURISTIC_COUNT:
                        break

            # B) Random Sampling
            needed = SAMPLES_PER_NODE - len(next_candidates)
            rem_list = list(remaining)
            if needed > 0 and rem_list:
                pool = [x for x in rem_list if x not in next_candidates]
                if len(pool) <= needed:
                    next_candidates.update(pool)

            for next_txn in next_candidates:
                new_seq = current_seq + [next_txn]
                new_cost = workload.get_opt_seq_cost(new_seq)

                new_rem = remaining.copy()
                new_rem.remove(next_txn)

                candidates_pool.append({
                    'cost': new_cost,
                    'seq': new_seq,
                    'rem': new_rem
                })

        if not candidates_pool:
            break

        candidates_pool.sort(key=lambda x: x['cost'])

        # Diversity Selection:
        # Take top K_BEST deterministically
        # Take remaining slots from a random sample of the top chunk to maintain diversity
        k_best = BEAM_WIDTH // 2
        k_random = BEAM_WIDTH - k_best

        new_beam = candidates_pool[:k_best]

        remaining_candidates = candidates_pool[k_best:min(len(candidates_pool), BEAM_WIDTH * 4)]
        if remaining_candidates and k_random > 0:
            if len(remaining_candidates) <= k_random:
                new_beam.extend(remaining_candidates)
            else:
                new_beam.extend(random.sample(remaining_candidates, k_random))

        beam = new_beam

    # 5. Final Result
    # Re-sort beam by pure cost
    beam.sort(key=lambda x: x['cost'])
    best_result = beam[0]
    return best_result['cost'], best_result['seq']
