        candidate = os.path.join(root, 'openevolve_examples', 'txn_scheduling')

def get_best_schedule(workload, num_seqs):
    """
    Hybrid Algorithm: Work-Density Beam Search + Adaptive Ruin-and-Recreate ILS.
    
    1. Constructive Phase: Beam Search.
       - Metric: Makespan - Gamma * Accumulated_Work.
       - Diversity: Maintains a mix of best-scoring and random candidates in the beam.
       
    2. Refinement Phase: Adaptive Iterated Local Search (ILS).
       - Primary Operator: Shift (Insert) & Swap.
       - Escape Mechanism: Ruin-and-Recreate (removes block, greedy re-insert).
       - Adaptive: Triggers 'Ruin' only when local search stagnates.

    Args:
        workload: Workload object
        num_seqs: Hint for computational budget

    Returns:
        Tuple of (lowest makespan, corresponding schedule)
    """

    # --- 0. Precomputation ---
    num_txns = workload.num_txns
    
    # Estimate transaction "work" (length)
    txn_lengths = {}
    for i in range(num_txns):
        try:
            # workload.txns[i][0] is metadata (id, type, ?, cost)
            txn_lengths[i] = workload.txns[i][0][3]
        except (IndexError, TypeError, AttributeError):
            txn_lengths[i] = 1.0

    # Sort indices by length (Longest Processing Time first)
    lpt_indices = sorted(txn_lengths.keys(), key=lambda k: txn_lengths[k], reverse=True)

    # --- 1. Constructive Phase: Beam Search ---
    
    BEAM_WIDTH = max(8, int(num_seqs * 1.2)) # Moderate width
    GAMMA = 1.5 # High bias towards completing work early
    
    # Initialization
    # Seed with top LPT transactions and some random ones
    seeds = set(lpt_indices[:BEAM_WIDTH])
    if len(seeds) < BEAM_WIDTH:
        remaining_slots = BEAM_WIDTH - len(seeds)
        seeds.update(random.sample(range(num_txns), min(num_txns - len(seeds), remaining_slots)))
    
    beam = []
    for t in seeds:
        seq = [t]
        cost = workload.get_opt_seq_cost(seq)
        acc_work = txn_lengths[t]
        score = cost - (GAMMA * acc_work)
        
        rem = set(range(num_txns))
        rem.remove(t)
        
        beam.append({
            'seq': seq,
            'cost': cost,
            'score': score,
            'acc_work': acc_work,
            'rem': rem
        })
    
    # Beam Loop
    for _ in range(num_txns - 1):
        candidates = []
        
        for parent in beam:
            rem_set = parent['rem']
            if not rem_set: continue
            
            # Select next transaction candidates
            to_eval = set()
            
            # A. Heuristic (LPT)
            count = 0
            for t in lpt_indices:
                if t in rem_set:
                    to_eval.add(t)
                    count += 1
                    if count >= 3: # Check top 3 longest available
                        break
            
            # B. Random (Diversity)
            rem_list = list(rem_set)
            # Add up to 3 random candidates
            sample_size = min(len(rem_list), 3)
            to_eval.update(random.sample(rem_list, sample_size))
            
            # Expand
            base_seq = parent['seq']
            base_work = parent['acc_work']
            
            for t in to_eval:
                new_seq = base_seq + [t]
                new_cost = workload.get_opt_seq_cost(new_seq)
                new_work = base_work + txn_lengths[t]
                new_score = new_cost - (GAMMA * new_work)
                
                new_rem = rem_set.copy()
                new_rem.remove(t)
                
                candidates.append({
                    'seq': new_seq,
                    'cost': new_cost,
                    'score': new_score,
                    'acc_work': new_work,
                    'rem': new_rem
                })
        
        if not candidates:
            break
            
        # Pruning Strategy
        # Sort by Score (Density)
        candidates.sort(key=lambda x: x['score'])
        
        # Keep best 60%
        k_best = int(BEAM_WIDTH * 0.6)
        next_beam = candidates[:k_best]
        
        # Randomly sample remaining slots from the top 3x pool
        # This maintains diversity better than just taking the next best
        remaining_needed = BEAM_WIDTH - len(next_beam)
        if remaining_needed > 0:
            pool = candidates[k_best : k_best * 4]
            if len(pool) <= remaining_needed:
                next_beam.extend(pool)
            else:
                next_beam.extend(random.sample(pool, remaining_needed))
        
        beam = next_beam

    # Extract best from construction
    beam.sort(key=lambda x: x['cost'])
    best_candidate = beam[0]
    
    current_schedule = list(best_candidate['seq'])
    current_cost = best_candidate['cost']
    
    # --- 2. Refinement Phase: Adaptive ILS ---
    
    best_schedule = list(current_schedule)
    best_cost = current_cost
    
    # Parameters
    MAX_ITER = 1200
    STAGNATION_LIMIT = 80 # Iterations without improvement to trigger Ruin
    
    temp = max(1.0, current_cost * 0.05)
    cooling = 0.99
    
    last_imp_iter = 0
    
    for it in range(MAX_ITER):
        
        # A. Check for Stagnation -> Ruin & Recreate
        if (it - last_imp_iter) > STAGNATION_LIMIT:
            # Ruin: Remove contiguous block
            # Size: 10-20%
            block_len = random.randint(2, max(3, int(num_txns * 0.2)))
            start_idx = random.randint(0, num_txns - block_len)
            
            working_seq = list(current_schedule)
            removed_block = working_seq[start_idx : start_idx + block_len]
            del working_seq[start_idx : start_idx + block_len]
            
            # Recreate: Greedy Insertion with limited sampling
            # Shuffle removed to avoid bias from original order
            random.shuffle(removed_block)
            
            for item in removed_block:
                best_pos = -1
                best_incr = float('inf')
                
                # Sampling positions to check
                # Always check 0, End, and near original position (approx)
                # plus some randoms
                candidates_idx = {0, len(working_seq)}
                
                # Random samples
                for _ in range(5):
                    candidates_idx.add(random.randint(0, len(working_seq)))
                
                for pos in candidates_idx:
                    working_seq.insert(pos, item)
                    c = workload.get_opt_seq_cost(working_seq)
                    if c < best_incr:
                        best_incr = c
                        best_pos = pos
                    del working_seq[pos]
                
                working_seq.insert(best_pos, item)
            
            # Update current to this new basin
            current_schedule = working_seq
            current_cost = workload.get_opt_seq_cost(current_schedule)
            
            # Reset stagnation counter (we just did a major move)
            last_imp_iter = it
            
            # Check if we got lucky
            if current_cost < best_cost:
                best_cost = current_cost
                best_schedule = list(current_schedule)
            
            continue # Skip normal SA step
            
        # B. Local Search (SA)
        neighbor = list(current_schedule)
        r = random.random()
        
        # Operators:
        # Shift (Insert) is king for scheduling. Give it high probability.
        if r < 0.65:
            # Shift
            idx1 = random.randint(0, num_txns - 1)
            val = neighbor.pop(idx1)
            idx2 = random.randint(0, num_txns - 1)
            neighbor.insert(idx2, val)
        elif r < 0.90:
            # Swap
            i, j = random.sample(range(num_txns), 2)
            neighbor[i], neighbor[j] = neighbor[j], neighbor[i]
        else:
            # Block Reverse (Local reordering)
            sz = random.randint(2, 6)
            start = random.randint(0, max(0, num_txns - sz))
            neighbor[start:start+sz] = reversed(neighbor[start:start+sz])
            
        n_cost = workload.get_opt_seq_cost(neighbor)
        delta = n_cost - current_cost
        
        accept = False
        if delta < 0:
            accept = True
            last_imp_iter = it # Improvement found
        elif temp > 0.001:
            if random.random() < math.exp(-delta / temp):
                accept = True
        
        if accept:
            current_schedule = neighbor
            current_cost = n_cost
            if current_cost < best_cost:
                best_cost = current_cost
                best_schedule = list(current_schedule)
                last_imp_iter = it
        
        temp *= cooling

    return best_cost, best_schedule


def get_random_costs():
    """
    Evaluate scheduling algorithm on three different workloads.
    
    Returns:
        Tuple of (total_makespan, list_of_schedules, execution_time)
    """
    start_time = time.time()
    
    # Workload 1: Complex mixed read/write transactions
    workload1 = Workload(WORKLOAD_1)
    makespan1, schedule1 = get_best_schedule(workload1, 10)
    cost1 = workload1.get_opt_seq_cost(schedule1)

    # Workload 2: Simple read-then-write pattern
    workload2 = Workload(WORKLOAD_2)
    makespan2, schedule2 = get_best_schedule(workload2, 10)
    cost2 = workload2.get_opt_seq_cost(schedule2)

    # Workload 3: Minimal read/write operations
    workload3 = Workload(WORKLOAD_3)
    makespan3, schedule3 = get_best_schedule(workload3, 10)
    cost3 = workload3.get_opt_seq_cost(schedule3)
    
    total_makespan = cost1 + cost2 + cost3
    schedules = [schedule1, schedule2, schedule3]
    execution_time = time.time() - start_time
    
    return total_makespan, schedules, execution_time
