2. Refined the mutation operators in LAHC to include Multi-Scale Block Inserts: separated "Micro Blocks" (size 2-4, 30%) for local adjustment and "Macro Blocks" (size 5-10, 10%) for structural changes, reducing pure random swaps to 10%.
3. Enhanced the "Kick" mechanism to perform a large block relocation plus local shuffle, providing a stronger perturbation to escape stagnation.
4. Tightened the randomization ranges for the Greedy phase parameters (alpha, threshold) to focus exploration around the high-performing baseline.

    # 4. LAHC Optimization with Kick and Cooling
    def run_lahc(schedule, initial_cost, budget, use_kick=False):
        current_seq = list(schedule)
        current_cost = initial_cost
        best_seq = list(schedule)
        best_cost = initial_cost

        history_len = 50
        history = [initial_cost] * history_len
        last_imp_idx = 0

        # Cooling start: last 20% of iterations are strict hill climbing
        cooling_start = int(budget * 0.8)

        for k in range(budget):
            # Kick Mechanism for stagnation in long runs (Marathon only)
            # Only applied before cooling phase
            if use_kick and k < cooling_start and (k - last_imp_idx > 500):
                # Major perturbation: Move a large block instead of just shuffle
                seg_len = random.randint(15, 30)
                if len(current_seq) > seg_len:
                    start_k = random.randint(0, len(current_seq) - seg_len)
                    block = current_seq[start_k : start_k + seg_len]
                    del current_seq[start_k : start_k + seg_len]

                    # Insert elsewhere
                    dest = random.randint(0, len(current_seq))
                    current_seq[dest : dest] = block

                    # Also slight shuffle within the block to mix
                    sub_start = dest
                    sub_end = dest + seg_len
                    if sub_end <= len(current_seq):
                        sub_segment = current_seq[sub_start:sub_end]
                        # Partial shuffle (swap a few pairs)
                        for _ in range(min(5, len(sub_segment)//2)):
                             s1, s2 = random.sample(range(len(sub_segment)), 2)
                             sub_segment[s1], sub_segment[s2] = sub_segment[s2], sub_segment[s1]
                        current_seq[sub_start:sub_end] = sub_segment

                    current_cost = workload.get_opt_seq_cost(current_seq)
                    history = [current_cost] * history_len
                    last_imp_idx = k
                    continue

            neighbor = list(current_seq)
            op_rand = random.random()

            # Operators:
            # 50% Single Insert
            # 30% Micro Block Insert (2-4 items)
            # 10% Macro Block Insert (5-10 items)
            # 10% Swap

            if op_rand < 0.50:
                # Single Insert
                idx1, idx2 = random.sample(range(len(neighbor)), 2)
                item = neighbor.pop(idx1)
                neighbor.insert(idx2, item)

            elif op_rand < 0.80:
                # Micro Block Insert (2-4 items)
                bsize = random.randint(2, 4)
                if len(neighbor) > bsize:
                    start = random.randint(0, len(neighbor) - bsize)
                    block = neighbor[start : start + bsize]
                    del neighbor[start : start + bsize]
                    dest = random.randint(0, len(neighbor))
                    neighbor[dest : dest] = block
                else:
                    continue

            elif op_rand < 0.90:
                # Macro Block Insert (5-10 items) - helps move dependent chains
                bsize = random.randint(5, 10)
                if len(neighbor) > bsize:
                    start = random.randint(0, len(neighbor) - bsize)
                    block = neighbor[start : start + bsize]
                    del neighbor[start : start + bsize]
                    dest = random.randint(0, len(neighbor))
                    neighbor[dest : dest] = block
                else:
                    continue

            else:
                # Swap
                idx1, idx2 = random.sample(range(len(neighbor)), 2)
                neighbor[idx1], neighbor[idx2] = neighbor[idx2], neighbor[idx1]

            new_cost = workload.get_opt_seq_cost(neighbor)

            # Acceptance Logic
            accepted = False

            # During cooling phase, enforce strict descent
            if k >= cooling_start:
                if new_cost <= current_cost:
                    accepted = True
            else:
                # Standard LAHC
                v = k % history_len
                if new_cost <= current_cost or new_cost <= history[v]:
                    accepted = True
                    # Update history
                    history[v] = current_cost

                current_cost = new_cost
                if current_cost < best_cost:
                    best_cost = current_cost
                    best_seq = list(current_seq)
                    last_imp_idx = k

        return best_cost, best_seq
