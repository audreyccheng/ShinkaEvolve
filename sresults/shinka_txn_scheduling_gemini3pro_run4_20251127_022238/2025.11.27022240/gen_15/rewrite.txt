def find_repo_root(start_path):
    """Find the repository root by looking for openevolve_examples directory."""
    current = os.path.abspath(start_path)
    # Search up the directory tree
    while current != os.path.dirname(current):  # Stop at filesystem root
        candidate = os.path.join(current, 'openevolve_examples', 'txn_scheduling')
        if os.path.exists(candidate):
            return current
        current = os.path.dirname(current)

    # If not found by searching up, try common locations relative to known paths
    # This handles when the program is copied to a results directory
    script_dir = os.path.dirname(os.path.abspath(__file__))
    possible_roots = [
        script_dir,  # Current directory
        os.path.dirname(script_dir),  # Parent
        os.path.dirname(os.path.dirname(script_dir)),  # Grandparent
        '/home/ubuntu/ShinkaEvolve',  # Absolute path fallback for Ubuntu
        '/Users/audreycc/Documents/Work/LLMTxn/ADRS-Exps/ShinkaEvolve',  # Absolute path fallback for macOS
    ]
    for root in possible_roots:
        candidate = os.path.join(root, 'openevolve_examples', 'txn_scheduling')
        if os.path.exists(candidate):
            return root

    raise RuntimeError(f"Could not find openevolve_examples directory. Searched from: {start_path}")

from txn_simulator import Workload
from workloads import WORKLOAD_1, WORKLOAD_2, WORKLOAD_3


def get_best_schedule(workload, num_seqs):
    """
    Get optimal schedule using Priority Greedy Construction and Iterated Local Search (ILS).
    
    Architecture:
    1.  **Length-Aware Greedy Phase**: Generates `num_seqs` candidate schedules. 
        Uses a large random sample for each step and breaks ties by selecting the 
        longest transaction (Best Fit Descending strategy).
    2.  **Initial Refinement**: Applies Adjacent Swap Descent to each candidate to 
        reach a local minimum.
    3.  **Selection**: Identifies the most promising schedule.
    4.  **Deep Optimization (ILS)**: Subjects the winner to multiple rounds of 
        perturbation (Insertion Kicks) and repair (Swap Descent) to escape local 
        optima that simple swapping cannot resolve.

    Args:
        workload: Workload object
        num_seqs: Number of initial greedy candidates to generate

    Returns:
        Tuple of (lowest makespan, corresponding schedule)
    """

    # --- Pre-computation: Transaction Lengths ---
    # Used as a heuristic tie-breaker. Longer transactions are harder to schedule,
    # so we prefer placing them when they fit well (cost doesn't increase).
    txn_lens = {}
    try:
        for i in range(workload.num_txns):
            raw_txn = workload.txns[i]
            if isinstance(raw_txn, (list, tuple)):
                raw_txn = raw_txn[0]
            txn_str = str(raw_txn)
            ops = len(re.findall(r'[rw]-\d+', txn_str))
            txn_lens[i] = ops
    except Exception:
        for i in range(workload.num_txns):
            txn_lens[i] = 1

    # Parameters
    SAMPLE_SIZE = 22       # Number of candidates to evaluate in greedy step
    ILS_ITERATIONS = 15    # Number of Kick-Repair cycles for the champion
    KICK_STRENGTH = 2      # Number of random insertion moves per kick

    candidates = []

    # --- Phase 1: Population Generation ---
    for _ in range(num_seqs):
        remaining = list(range(workload.num_txns))
        
        # Start with a random transaction to ensure diversity
        start_txn = random.choice(remaining)
        current_seq = [start_txn]
        remaining.remove(start_txn)
        
        # Greedy Construction
        while remaining:
            # Sample random candidates
            sample_len = min(len(remaining), SAMPLE_SIZE)
            sample_cands = random.sample(remaining, sample_len)
            
            best_cand = -1
            # Optimization Goal: Minimize Cost, then Maximize Length (Minimize -Length)
            best_score = (float('inf'), 0)
            
            for t in sample_cands:
                # Check cost of appending t
                test_seq = current_seq + [t]
                cost = workload.get_opt_seq_cost(test_seq)
                
                # Tie-breaker logic
                score = (cost, -txn_lens[t])
                
                if score < best_score:
                    best_score = score
                    best_cand = t
            
            current_seq.append(best_cand)
            remaining.remove(best_cand)
        
        # Fast Polish: Adjacent Swap Descent
        # Bring this candidate to its local minimum before comparing
        current_cost = workload.get_opt_seq_cost(current_seq)
        improved = True
        while improved:
            improved = False
            for i in range(len(current_seq) - 1):
                # Swap
                current_seq[i], current_seq[i+1] = current_seq[i+1], current_seq[i]
                new_cost = workload.get_opt_seq_cost(current_seq)
                
                if new_cost < current_cost:
                    current_cost = new_cost
                    improved = True
                else:
                    # Revert
                    current_seq[i], current_seq[i+1] = current_seq[i+1], current_seq[i]
        
        candidates.append((current_cost, current_seq))

    # --- Phase 2: Champion Selection ---
    candidates.sort(key=lambda x: x[0])
    best_overall_cost, best_overall_schedule = candidates[0]

    # --- Phase 3: Deep Optimization (ILS) ---
    # Apply perturbations (Insertion) to escape swap-based local optima
    
    current_seq = list(best_overall_schedule)
    current_cost = best_overall_cost
    
    for _ in range(ILS_ITERATIONS):
        # Save state before kick
        backup_seq = list(current_seq)
        
        # A. Kick: Random Insertion Perturbation
        # Move transactions to random positions to shake up dependencies
        for _ in range(KICK_STRENGTH):
            if len(current_seq) < 2: break
            idx_from = random.randint(0, len(current_seq) - 1)
            idx_to = random.randint(0, len(current_seq) - 1)
            
            if idx_from != idx_to:
                txn = current_seq.pop(idx_from)
                current_seq.insert(idx_to, txn)
        
        # B. Repair: Adjacent Swap Descent
        # Re-optimize the perturbed schedule
        repair_improved = True
        repair_cost = workload.get_opt_seq_cost(current_seq)
        
        while repair_improved:
            repair_improved = False
            for i in range(len(current_seq) - 1):
                current_seq[i], current_seq[i+1] = current_seq[i+1], current_seq[i]
                new_cost = workload.get_opt_seq_cost(current_seq)
                
                if new_cost < repair_cost:
                    repair_cost = new_cost
                    repair_improved = True
                else:
                    current_seq[i], current_seq[i+1] = current_seq[i+1], current_seq[i]
        
        # C. Acceptance Criterion
        # Simple greedy acceptance: if better than global best, update.
        # Note: In standard ILS we might accept worse solutions to explore, 
        # but here we stick to greedy descent on the global best to ensure monotonicity.
        if repair_cost < best_overall_cost:
            best_overall_cost = repair_cost
            best_overall_schedule = list(current_seq)
            # Update current to the new best
            current_cost = repair_cost
        else:
            # Revert to backup (Exploit the basin of the best found so far)
            current_seq = backup_seq

    return best_overall_cost, best_overall_schedule


def get_random_costs():
    """
    Evaluate scheduling algorithm on three different workloads.

    Returns:
        Tuple of (total_makespan, list_of_schedules, execution_time)
    """
    start_time = time.time()
    
    # We pass 10 sequences to generation phase. 
    # The algorithm picks the best of 10 and then runs deep optimization.
    NUM_SEQS = 10

    # Workload 1: Complex mixed read/write transactions
    workload = Workload(WORKLOAD_1)
    makespan1, schedule1 = get_best_schedule(workload, NUM_SEQS)
    cost1 = workload.get_opt_seq_cost(schedule1)

    # Workload 2: Simple read-then-write pattern
    workload2 = Workload(WORKLOAD_2)
    makespan2, schedule2 = get_best_schedule(workload2, NUM_SEQS)
    cost2 = workload2.get_opt_seq_cost(schedule2)

    # Workload 3: Minimal read/write operations
    workload3 = Workload(WORKLOAD_3)
    makespan3, schedule3 = get_best_schedule(workload3, NUM_SEQS)
    cost3 = workload3.get_opt_seq_cost(schedule3)

    total_makespan = cost1 + cost2 + cost3
    schedules = [schedule1, schedule2, schedule3]
    execution_time = time.time() - start_time

    return total_makespan, schedules, execution_time
