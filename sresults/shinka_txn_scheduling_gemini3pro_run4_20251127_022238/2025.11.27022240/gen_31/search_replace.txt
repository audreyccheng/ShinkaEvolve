<DESCRIPTION>
Replaces the greedy construction phase in `get_best_schedule` with a Weighted Score Greedy strategy.
- Introduces `alpha` parameter to trade off immediate makespan cost vs transaction duration (heuristic).
- `alpha` is randomized for each sequence generation to produce diverse high-quality candidates.
- Adjusts "Big Rocks" parameters: increased pool size (10) and lowered threshold (85%) to consider more bottlenecks.
- The 0-th iteration uses a deterministic approach with the longest transaction to ensure a strong baseline.
- Retains the Multi-Candidate LAHC refinement strategy.
</DESCRIPTION>

    Get optimal schedule using Multi-Candidate Greedy Construction and LAHC Refinement.

    Args:
        workload: Workload object containing transaction data
        num_seqs: Number of sequences to sample for greedy selection

    Returns:
        Tuple of (lowest makespan, corresponding schedule)
    """

    # Compute transaction durations for heuristic guidance
    txn_durations = {t: workload.get_opt_seq_cost([t]) for t in range(workload.num_txns)}

    sorted_txns_by_len = sorted(range(workload.num_txns), key=lambda t: txn_durations[t], reverse=True)

    for i in range(num_seqs):

        # Iteration >0: Randomized with Dynamic Big Rocks heuristic

        start_txn = random.randint(0, workload.num_txns - 1)
        txn_seq = [start_txn]

        remaining_txns = list(range(workload.num_txns))
        remaining_txns.remove(start_txn)

        while remaining_txns:
            candidates = set()
            remaining_set = set(remaining_txns)

            if i == 0:
                # Full scan for first sequence
                candidates = remaining_set
            else:
                # Dynamic Big Rocks Heuristic
                # Identify transactions that are "large" relative to what's left
                max_dur = -1
                for t in sorted_txns_by_len:
                    if t in remaining_set:
                        max_dur = txn_durations[t]
                        break

                # Dynamic threshold: 90% of the longest remaining transaction
                threshold = max_dur * 0.90
                added_rocks = 0
                max_rocks = 6 # Cap to ensure performance

                # Add Big Rocks
                for t in sorted_txns_by_len:
                    if t in remaining_set:
                        if txn_durations[t] >= threshold:
                            candidates.add(t)
                            added_rocks += 1
                            if added_rocks >= max_rocks:
                                break
                        else:
                            # Sorted list, so we can stop early
                            break

            best_t = -1

                    best_metric = metric
                    best_t = t

            remaining_txns.remove(best_t)

        total_cost = workload.get_opt_seq_cost(txn_seq)
        candidates_pool.append((total_cost, txn_seq))

def get_best_schedule(workload, num_seqs):
    """
    Get optimal schedule using Multi-Candidate Weighted Greedy Construction and LAHC Refinement.

    Args:
        workload: Workload object containing transaction data
        num_seqs: Number of sequences to sample for greedy selection

    Returns:
        Tuple of (lowest makespan, corresponding schedule)
    """
    # 1. Pre-calculation
    # Compute transaction durations for heuristic guidance
    txn_durations = {t: workload.get_opt_seq_cost([t]) for t in range(workload.num_txns)}
    # Sort transactions by duration (descending) for Big Rocks lookup
    sorted_txns_by_len = sorted(range(workload.num_txns), key=lambda t: txn_durations[t], reverse=True)

    candidates_pool = []

    # 2. Greedy Construction Phase
    for i in range(num_seqs):
        # Iteration 0: Deterministic baseline with lowest alpha (cost priority)
        # Iteration >0: Randomized alpha for diversity

        # Alpha controls the weight of transaction duration in selection
        # Score = Cost - (alpha * Duration)
        # Higher alpha = prefer longer transactions even if they increase makespan
        if i == 0:
            alpha = 0.001
            start_txn = sorted_txns_by_len[0] # Deterministic start
        else:
            alpha = random.uniform(0.0, 0.4)
            start_txn = random.randint(0, workload.num_txns - 1)

        txn_seq = [start_txn]
        remaining_txns = list(range(workload.num_txns))
        remaining_txns.remove(start_txn)

        # Optimization: maintain set for fast lookup
        remaining_set = set(remaining_txns)

        while remaining_txns:
            candidates = set()

            if i == 0:
                # Full scan for first sequence
                candidates = remaining_set
            else:
                # Dynamic Big Rocks Heuristic
                # Find max duration among remaining
                max_dur = 0
                for t in sorted_txns_by_len:
                    if t in remaining_set:
                        max_dur = txn_durations[t]
                        break

                # Dynamic threshold: include transactions close to the max remaining length
                threshold = max_dur * 0.85
                added_rocks = 0
                max_rocks = 10

                for t in sorted_txns_by_len:
                    if t in remaining_set:
                        if txn_durations[t] >= threshold:
                            candidates.add(t)
                            added_rocks += 1
                            if added_rocks >= max_rocks:
                                break
                        else:
                            break

                # Fill rest with random samples
                target_total = 25
                if len(remaining_set) <= target_total:
                     candidates = remaining_set
                else:
                    while len(candidates) < target_total:
                        # Random sampling from list is O(1)
                        t = random.choice(remaining_txns)
                        candidates.add(t)

            # Selection: Minimize Weighted Score
            best_t = -1
            best_score = float('inf')

            for t in candidates:
                cost = workload.get_opt_seq_cost(txn_seq + [t])

                # Weighted score
                score = cost - (alpha * txn_durations[t])

                if score < best_score:
                    best_score = score
                    best_t = t
                elif score == best_score:
                    if txn_durations[t] > txn_durations.get(best_t, 0):
                        best_t = t

            txn_seq.append(best_t)
            remaining_txns.remove(best_t)
            remaining_set.remove(best_t)

    seen_costs = set()

            seen_costs.add(cost)
        if len(unique_candidates) >= 3:
            break

    best_overall_schedule = []
