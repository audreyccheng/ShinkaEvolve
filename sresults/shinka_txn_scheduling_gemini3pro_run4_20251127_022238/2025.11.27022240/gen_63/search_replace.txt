<DESCRIPTION>
Implement a Funnel Architecture for scheduling optimization:
1.  **Diverse Greedy Construction**: Uses randomized parameters (alpha, rock_limit) to generate structurally diverse initial candidates. It employs a weighted scoring function (`cost - alpha * duration`) to prioritize "big rocks" (long transactions) early.
2.  **Sprint Phase**: Runs a short Late Acceptance Hill Climbing (LAHC) search on the top unique candidates to filter and polish them.
3.  **Marathon Phase**: Runs a deep LAHC search on the best candidate.
4.  **Enhanced Operators**: The LAHC includes Insert (50%), Block Move (40%), and Swap (10%) operators.
5.  **Stagnation Kick**: In the Marathon phase, if no improvement is found for a long period, a segment of the schedule is shuffled to escape local optima.

def get_best_schedule(workload, num_seqs):
    """
    Get optimal schedule using Diverse Weighted Greedy Construction and Funnel LAHC Refinement.

    Phases:
    1. Diverse Greedy Construction: Randomizes scoring parameters to find different high-quality starting points.
    2. Sprint Phase: Short optimization on top candidates.
    3. Marathon Phase: Deep optimization with stagnation kick on the best candidate.
    """
    # 1. Pre-calculation
    txn_durations = {t: workload.get_opt_seq_cost([t]) for t in range(workload.num_txns)}
    sorted_txns_by_len = sorted(range(workload.num_txns), key=lambda t: txn_durations[t], reverse=True)

    candidates_pool = []

    # 2. Greedy Construction Loop
    for i in range(num_seqs):
        # Diversity: Randomize parameters
        if i == 0:
            # Baseline (Conservative)
            alpha = 0.4
            rock_threshold_ratio = 0.90
            rock_limit = 5
        else:
            # Exploratory
            alpha = random.uniform(0.2, 0.6)
            rock_threshold_ratio = random.uniform(0.85, 0.95)
            rock_limit = random.randint(3, 8)

        # Start with random transaction
        start_txn = random.randint(0, workload.num_txns - 1)
        schedule = [start_txn]
        remaining = set(range(workload.num_txns))
        remaining.remove(start_txn)

        while remaining:
            candidates = set()

            # A. Big Rocks Heuristic
            # Efficiently find max duration in remaining
            max_dur = 0
            for t in sorted_txns_by_len:
                if t in remaining:
                    max_dur = txn_durations[t]
                    break

            if max_dur > 0:
                threshold = max_dur * rock_threshold_ratio
                added = 0
                for t in sorted_txns_by_len:
                    if t in remaining:
                        if txn_durations[t] >= threshold:
                            candidates.add(t)
                            added += 1
                            if added >= rock_limit:
                                break
                        else:
                            break # List is sorted

            # B. Random Sampling (for diversity in large sets)
            if len(remaining) > 20:
                needed = 15
                current_count = len(candidates)
                if current_count < needed:
                    pool = list(remaining)
                    # Sample random subset
                    candidates.update(random.sample(pool, min(len(pool), needed - current_count + 5)))
            else:
                candidates.update(remaining)

            # C. Weighted Scoring
            # Score = Cost - (Alpha * Duration)
            best_t = -1
            best_score = float('inf')

            for t in candidates:
                cost = workload.get_opt_seq_cost(schedule + [t])
                score = cost - (alpha * txn_durations[t])

                if score < best_score:
                    best_score = score
                    best_t = t
                elif score == best_score:
                    if txn_durations[t] > txn_durations.get(best_t, 0):
                        best_t = t

            schedule.append(best_t)
            remaining.remove(best_t)

        cost = workload.get_opt_seq_cost(schedule)
        candidates_pool.append((cost, schedule))

    # 3. Refinement Phase (Funnel Architecture)

    # Sort candidates
    candidates_pool.sort(key=lambda x: x[0])

    # Select top 3 unique candidates
    unique_candidates = []
    seen_hashes = set()
    for cost, seq in candidates_pool:
        h = tuple(seq)
        if h not in seen_hashes:
            unique_candidates.append((cost, list(seq)))
            seen_hashes.add(h)
        if len(unique_candidates) >= 3:
            break

    if not unique_candidates:
        unique_candidates = [candidates_pool[0]]

    # Helper: Late Acceptance Hill Climbing
    def run_lahc(schedule, start_cost, budget, enable_kick=False):
        current_seq = list(schedule)
        current_cost = start_cost
        best_seq = list(schedule)
        best_cost = start_cost

        history_len = 50
        history = [start_cost] * history_len
        last_imp_idx = 0

        for k in range(budget):
            # Kick (Perturbation) Logic
            if enable_kick and (k - last_imp_idx > 500):
                # Shuffle a random segment
                seg_len = random.randint(10, 20)
                if len(current_seq) > seg_len:
                    start = random.randint(0, len(current_seq) - seg_len)
                    segment = current_seq[start : start + seg_len]
                    random.shuffle(segment)
                    current_seq[start : start + seg_len] = segment

                    current_cost = workload.get_opt_seq_cost(current_seq)
                    history = [current_cost] * history_len
                    last_imp_idx = k
                    continue

            neighbor = list(current_seq)
            op_rand = random.random()

            # Operators: 50% Insert, 40% Block Move, 10% Swap
            if op_rand < 0.50:
                # Insert
                idx1 = random.randint(0, len(neighbor) - 1)
                idx2 = random.randint(0, len(neighbor) - 1)
                if idx1 != idx2:
                    item = neighbor.pop(idx1)
                    neighbor.insert(idx2, item)
            elif op_rand < 0.90:
                # Block Move (Size 2-6)
                bsize = random.randint(2, 6)
                if len(neighbor) > bsize:
                    start = random.randint(0, len(neighbor) - bsize)
                    block = neighbor[start : start + bsize]
                    del neighbor[start : start + bsize]
                    dest = random.randint(0, len(neighbor))
                    neighbor[dest : dest] = block
                else:
                    continue
            else:
                # Swap
                idx1, idx2 = random.sample(range(len(neighbor)), 2)
                neighbor[idx1], neighbor[idx2] = neighbor[idx2], neighbor[idx1]

            new_cost = workload.get_opt_seq_cost(neighbor)

            # Acceptance
            v = k % history_len
            if new_cost <= current_cost or new_cost <= history[v]:
                current_seq = neighbor
                current_cost = new_cost
                if current_cost < best_cost:
                    best_cost = current_cost
                    best_seq = list(current_seq)
                    last_imp_idx = k

            history[v] = current_cost

        return best_cost, best_seq

    # Sprint Phase (Medium optimization on top candidates)
    sprint_results = []
    for cost, seq in unique_candidates:
        c, s = run_lahc(seq, cost, 300, enable_kick=False)
        sprint_results.append((c, s))

    sprint_results.sort(key=lambda x: x[0])

    # Marathon Phase (Deep optimization on winner)
    winner_cost, winner_seq = sprint_results[0]
    final_cost, final_seq = run_lahc(winner_seq, winner_cost, 2500, enable_kick=True)

    return final_cost, final_seq
