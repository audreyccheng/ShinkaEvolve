def get_best_schedule(workload, num_seqs):
    """
    Optimized Scheduler using Variable-Strategy Greedy and Deep LAHC Refinement.
    
    Architecture:
    1.  **Metric Analysis**: Pre-compute transaction lengths (ops count).
    2.  **Diverse Greedy Phase**: Generate `num_seqs` candidates using a variable `alpha` parameter.
        -   `alpha` controls the heuristic: Low alpha favors filling holes (Shortest/Best Fit),
            High alpha favors scheduling long tasks early (Longest Job First).
        -   This ensures the candidate pool covers different structural strategies.
    3.  **Sprint Phase**: Run short Late Acceptance Hill Climbing (LAHC) on the top 4 distinct greedy candidates.
    4.  **Marathon Phase**: Run extended LAHC on the single best candidate.
        -   Operators: Single Insertion, Block Insertion (size 2-5), Adjacent Swap.
    """
    
    # --- Pre-computation ---
    txn_lens = {}
    try:
        for i in range(workload.num_txns):
            raw_txn = workload.txns[i]
            if isinstance(raw_txn, (list, tuple)):
                raw_txn = raw_txn[0]
            txn_str = str(raw_txn)
            ops = len(re.findall(r'[rw]-\d+', txn_str))
            txn_lens[i] = ops
    except Exception:
        for i in range(workload.num_txns):
            txn_lens[i] = 1

    # --- LAHC Helper Function ---
    def run_lahc(start_seq, start_cost, iterations, history_len):
        curr_s = list(start_seq)
        curr_c = start_cost
        best_s = list(start_seq)
        best_c = start_cost
        
        history = [curr_c] * history_len
        
        for i in range(iterations):
            # Mutation
            neigh_s = list(curr_s)
            slen = len(neigh_s)
            r = random.random()
            
            if r < 0.5:
                # Single Shift (Insertion) - 50%
                if slen < 2: continue
                f = random.randint(0, slen-1)
                t = random.randint(0, slen-1)
                if f == t: continue
                item = neigh_s.pop(f)
                neigh_s.insert(t, item)
                
            elif r < 0.85:
                # Block Shift - 35%
                # Moves a contiguous block to a new location.
                # Critical for moving dependent chains of transactions together.
                if slen < 5: continue
                # Allow slightly larger blocks (2 to 5)
                bsize = random.randint(2, 5)
                f = random.randint(0, slen - bsize)
                block = neigh_s[f : f+bsize]
                del neigh_s[f : f+bsize]
                t = random.randint(0, len(neigh_s))
                neigh_s[t:t] = block
                
            else:
                # Adjacent Swap - 15%
                if slen < 2: continue
                idx = random.randint(0, slen - 2)
                neigh_s[idx], neigh_s[idx+1] = neigh_s[idx+1], neigh_s[idx]
            
            # Evaluation
            neigh_c = workload.get_opt_seq_cost(neigh_s)
            
            # Late Acceptance Logic
            h_idx = i % history_len
            if neigh_c <= curr_c or neigh_c <= history[h_idx]:
                curr_s = neigh_s
                curr_c = neigh_c
                if curr_c < best_c:
                    best_c = curr_c
                    best_s = list(curr_s)
            
            history[h_idx] = curr_c
            
        return best_c, best_s

    # --- Phase 1: Variable-Strategy Greedy ---
    candidates = []
    GREEDY_SAMPLE = 8
    # Alphas: 0.0=Fit Holes, 2.0=Force Longest. 
    # We cycle through these to get diversity in the population.
    strategy_alphas = [0.0, 0.2, 0.5, 1.0, 1.5, 2.0]
    
    for i in range(num_seqs):
        alpha = strategy_alphas[i % len(strategy_alphas)]
        
        remaining = set(range(workload.num_txns))
        
        # Start Selection: High alpha -> start with big rock. Low alpha -> random.
        if alpha >= 1.0:
            # Pick one of top 5 longest
            top_txns = sorted(list(remaining), key=lambda x: txn_lens[x], reverse=True)[:5]
            start_txn = random.choice(top_txns)
        else:
            start_txn = random.choice(list(remaining))
            
        current_seq = [start_txn]
        remaining.remove(start_txn)
        
        while remaining:
            # Adaptive Big Rocks Pool
            # Recalculate max length in remaining set
            rem_list = list(remaining)
            max_len = 0
            for t in rem_list:
                if txn_lens[t] > max_len: max_len = txn_lens[t]
            
            threshold = max_len * 0.85
            big_rocks = [t for t in rem_list if txn_lens[t] >= threshold]
            
            # Pool Construction
            pool = big_rocks[:5] # Take top 5 big rocks
            
            if len(pool) < GREEDY_SAMPLE and len(rem_list) > len(pool):
                needed = GREEDY_SAMPLE - len(pool)
                others = [t for t in rem_list if t not in pool]
                if others:
                    pool.extend(random.sample(others, min(len(others), needed)))
            
            # Evaluation
            best_t = -1
            best_score = float('inf')
            
            for t in pool:
                cost = workload.get_opt_seq_cost(current_seq + [t])
                # Weighted Heuristic
                score = cost - (alpha * txn_lens[t])
                
                if score < best_score:
                    best_score = score
                    best_t = t
            
            current_seq.append(best_t)
            remaining.remove(best_t)
            
        # Quick Polish (First Improvement Descent, Limited Passes)
        # We limit passes to ensure we don't spend too much time here
        curr_cost = workload.get_opt_seq_cost(current_seq)
        for _ in range(2): # Max 2 passes
            improved = False
            for idx in range(len(current_seq) - 1):
                current_seq[idx], current_seq[idx+1] = current_seq[idx+1], current_seq[idx]
                new_cost = workload.get_opt_seq_cost(current_seq)
                if new_cost < curr_cost:
                    curr_cost = new_cost
                    improved = True
                else:
                    # Revert
                    current_seq[idx], current_seq[idx+1] = current_seq[idx+1], current_seq[idx]
            if not improved:
                break
                
        candidates.append((curr_cost, current_seq))

    # --- Phase 2: Sprint (Short LAHC) ---
    candidates.sort(key=lambda x: x[0])
    
    # Select top 4 distinct schedules
    sprint_cands = []
    seen_costs = set()
    for c, s in candidates:
        if c not in seen_costs:
            sprint_cands.append((c, list(s)))
            seen_costs.add(c)
        if len(sprint_cands) >= 4:
            break
            
    if not sprint_cands:
        sprint_cands = [candidates[0]]
        
    sprint_results = []
    for c, s in sprint_cands:
        # 300 iterations, history 50
        res = run_lahc(s, c, iterations=300, history_len=50)
        sprint_results.append(res)
        
    # --- Phase 3: Marathon (Deep LAHC) ---
    sprint_results.sort(key=lambda x: x[0])
    champion_cost, champion_seq = sprint_results[0]
    
    # 3000 iterations, history 60
    final_cost, final_seq = run_lahc(champion_seq, champion_cost, iterations=3000, history_len=60)

    return final_cost, final_seq

def get_random_costs():
    """
    Evaluate scheduling algorithm on three different workloads.

    Returns:
        Tuple of (total_makespan, list_of_schedules, execution_time)
    """
    start_time = time.time()
    NUM_SEQS = 10 

    workload = Workload(WORKLOAD_1)
    makespan1, schedule1 = get_best_schedule(workload, NUM_SEQS)
    cost1 = workload.get_opt_seq_cost(schedule1)

    workload2 = Workload(WORKLOAD_2)
    makespan2, schedule2 = get_best_schedule(workload2, NUM_SEQS)
    cost2 = workload2.get_opt_seq_cost(schedule2)

    workload3 = Workload(WORKLOAD_3)
    makespan3, schedule3 = get_best_schedule(workload3, NUM_SEQS)
    cost3 = workload3.get_opt_seq_cost(schedule3)

    total_makespan = cost1 + cost2 + cost3
    schedules = [schedule1, schedule2, schedule3]
    execution_time = time.time() - start_time

    return total_makespan, schedules, execution_time
