<DESCRIPTION>
Replaces the basic greedy + random search with a sophisticated "Funnel Refinement" strategy.
1. **Diverse Greedy Construction**: Randomizes the `alpha` parameter (0.2-0.6) and Big Rocks threshold per sequence to create a structurally diverse initial pool.
2. **Funnel Stages**:
   - **Warmup**: Runs 50 LAHC iterations on all greedy candidates to filter out unstable ones (greedy cost is often misleading).
   - **Sprint**: Runs 400 LAHC iterations on the top 3 unique candidates.
   - **Marathon**: Runs 2500 LAHC iterations on the final winner.
3. **Enhanced LAHC**: Uses weighted operators (50% Insert, 40% Block Insert up to 8 items, 10% Swap) and a "Kick" mechanism (shuffling a segment) to escape stagnation in the Marathon phase.
</DESCRIPTION>

    Args:
        workload: Workload object containing transaction data
        num_seqs: Number of sequences to sample for greedy selection

    Returns:
        Tuple of (lowest makespan, corresponding schedule)

            added_long = 0
            for t in sorted_txns_by_len:
                if t in remaining_set:
                    candidates.add(t)
                    added_long += 1
                    if added_long >= 3:
                        break

            target_samples = 20

            max_attempts = target_samples * 2

def get_best_schedule(workload, num_seqs):
    """
    Get optimal schedule using Funnel Refinement on Diverse Greedy Candidates.

    Phases:
    1. Greedy Construction: Generates candidates using randomized Weighted Greedy heuristic.
    2. Funnel Refinement:
       - Warmup: Short LAHC on all candidates to measure stability.
       - Sprint: Medium LAHC on top 3 candidates.
       - Marathon: Long LAHC on the best candidate with Stagnation Kick.
    """
    # Precompute transaction durations
    txn_durations = {t: workload.get_opt_seq_cost([t]) for t in range(workload.num_txns)}
    sorted_txns_by_len = sorted(range(workload.num_txns), key=lambda t: txn_durations[t], reverse=True)

    candidates_pool = []

    # 1. Greedy Phase (Diversity-Driven)
    for i in range(num_seqs):
        # Randomize parameters for structural diversity
        if i == 0:
            # Baseline parameters
            alpha = 0.4
            rock_threshold_ratio = 0.90
        else:
            alpha = random.uniform(0.2, 0.6)
            rock_threshold_ratio = random.uniform(0.85, 0.98)

        # Start with random transaction
        start_txn = random.randint(0, workload.num_txns - 1)
        schedule = [start_txn]
        remaining = set(range(workload.num_txns))
        remaining.remove(start_txn)

        while remaining:
            candidates = set()

            # Big Rocks Strategy (Dynamic Threshold)
            max_dur_remaining = 0
            for t in sorted_txns_by_len:
                if t in remaining:
                    max_dur_remaining = txn_durations[t]
                    break

            threshold = max_dur_remaining * rock_threshold_ratio

            rocks_added = 0
            for t in sorted_txns_by_len:
                if t in remaining:
                    if txn_durations[t] >= threshold:
                        candidates.add(t)
                        rocks_added += 1
                        if rocks_added >= 6: # Cap rocks to avoid checking too many
                            break
                    else:
                        break # Sorted by length, so rest are smaller

            # Random Sampling (Exploration)
            # Full scan for the first baseline iteration
            if i == 0:
                candidates = remaining
            else:
                pool_list = list(remaining)
                if len(pool_list) > 20:
                    candidates.update(random.sample(pool_list, 15))
                else:
                    candidates.update(remaining)

            # Weighted Selection
            # Score = NewMakespan - (Alpha * TxnDuration)
            best_t = -1
            best_score = float('inf')

            for t in candidates:
                cost = workload.get_opt_seq_cost(schedule + [t])
                score = cost - (alpha * txn_durations[t])

                if score < best_score:
                    best_score = score
                    best_t = t
                elif score == best_score:
                    # Tie-break: Prefer longer duration
                    if txn_durations[t] > txn_durations.get(best_t, 0):
                        best_t = t

            schedule.append(best_t)
            remaining.remove(best_t)

        initial_cost = workload.get_opt_seq_cost(schedule)
        candidates_pool.append((initial_cost, schedule))

    # 2. Late Acceptance Hill Climbing (LAHC) Helper
    def run_lahc(schedule, start_cost, budget, enable_kick=False):
        current_seq = list(schedule)
        current_cost = start_cost
        best_seq = list(schedule)
        best_cost = start_cost

        history_len = 50
        history = [start_cost] * history_len
        last_imp_idx = 0

        for k in range(budget):
            # Kick logic for stagnation (Marathon phase)
            if enable_kick and (k - last_imp_idx > 600):
                # Shuffle a sub-segment
                seg_len = random.randint(10, 20)
                if len(current_seq) > seg_len:
                    start_idx = random.randint(0, len(current_seq) - seg_len)
                    segment = current_seq[start_idx : start_idx + seg_len]
                    random.shuffle(segment)
                    current_seq[start_idx : start_idx + seg_len] = segment

                    # Reset after kick
                    current_cost = workload.get_opt_seq_cost(current_seq)
                    history = [current_cost] * history_len
                    last_imp_idx = k
                    continue

            neighbor = list(current_seq)
            op_rand = random.random()

            # Operators: 50% Insert, 40% Block Insert, 10% Swap
            if op_rand < 0.50:
                # Single Insert
                idx1, idx2 = random.sample(range(len(neighbor)), 2)
                item = neighbor.pop(idx1)
                neighbor.insert(idx2, item)

            elif op_rand < 0.90:
                # Block Insert (Move contiguous chunk of 2-8 items)
                # Helps move dependency chains together
                bsize = random.randint(2, 8)
                if len(neighbor) > bsize:
                    start = random.randint(0, len(neighbor) - bsize)
                    block = neighbor[start : start + bsize]
                    del neighbor[start : start + bsize]
                    dest = random.randint(0, len(neighbor))
                    neighbor[dest : dest] = block
                else:
                    continue
            else:
                # Swap (Destructive but useful for jumps)
                idx1, idx2 = random.sample(range(len(neighbor)), 2)
                neighbor[idx1], neighbor[idx2] = neighbor[idx2], neighbor[idx1]

            new_cost = workload.get_opt_seq_cost(neighbor)

            # LAHC Acceptance
            v = k % history_len
            if new_cost <= current_cost or new_cost <= history[v]:
                current_seq = neighbor
                current_cost = new_cost
                if current_cost < best_cost:
                    best_cost = current_cost
                    best_seq = list(current_seq)
                    last_imp_idx = k

            history[v] = current_cost

        return best_cost, best_seq

    # 3. Funnel Refinement Process

    # Stage 1: Warmup (Filter unstable candidates)
    # Run short LAHC on all candidates to get a better cost estimate
    warmup_candidates = []
    for cost, seq in candidates_pool:
        wc, ws = run_lahc(seq, cost, 50, enable_kick=False)
        warmup_candidates.append((wc, ws))

    # Stage 2: Selection
    warmup_candidates.sort(key=lambda x: x[0])

    unique_candidates = []
    seen_hashes = set()
    for cost, seq in warmup_candidates:
        h = tuple(seq)
        if h not in seen_hashes:
            unique_candidates.append((cost, list(seq)))
            seen_hashes.add(h)
        if len(unique_candidates) >= 3:
            break

    if not unique_candidates:
        unique_candidates = [warmup_candidates[0]]

    # Stage 3: Sprint (Medium optimization on top candidates)
    sprint_results = []
    for cost, seq in unique_candidates:
        sc, ss = run_lahc(seq, cost, 400, enable_kick=False)
        sprint_results.append((sc, ss))

    # Stage 4: Marathon (Long optimization on winner)
    sprint_results.sort(key=lambda x: x[0])
    winner_cost, winner_seq = sprint_results[0]

    final_cost, final_seq = run_lahc(winner_seq, winner_cost, 2500, enable_kick=True)

    return final_cost, final_seq
