def get_best_schedule(workload, num_seqs):

    1. Diversity-Driven Greedy: Randomizes alpha and Big Rocks threshold.

        if i == 0:

        remaining_txns.remove(start_txn)

                candidates = remaining_txns
            else:
                # Dynamic Big Rocks Heuristic
                # Find max duration in remaining set
                max_dur = 0

                    if t in remaining_txns:

                    best_t = t

def get_best_schedule(workload, num_seqs):
    """
    Get optimal schedule using Decaying-Alpha Greedy Construction and Multi-Scale LAHC.

    Strategies:
    1. Decaying Alpha Greedy: Prioritizes long transactions early (high alpha), focuses on fit later (low alpha).
    2. Multi-Scale LAHC: Uses micro and macro block moves to escape optima.
    3. Final Descent: Switches to strict hill climbing at the end of the budget.
    """
    # 1. Pre-calculation
    # Compute transaction durations for heuristic guidance
    txn_durations = {t: workload.get_opt_seq_cost([t]) for t in range(workload.num_txns)}
    # Sort transactions by duration (descending) for Big Rocks lookup
    sorted_txns_by_len = sorted(range(workload.num_txns), key=lambda t: txn_durations[t], reverse=True)

    candidates_pool = []

    # 2. Greedy Construction Phase
    for i in range(num_seqs):
        # Decaying Alpha parameters
        # Start high (prioritize duration) -> End low (prioritize cost fit)
        start_alpha = 0.6
        end_alpha = 0.1

        # Diversity: Perturb parameters slightly for different candidates
        if i > 0:
            start_alpha = random.uniform(0.5, 0.7)
            end_alpha = random.uniform(0.0, 0.2)

        # Determine start transaction
        if i == 0:
            start_txn = sorted_txns_by_len[0] # Deterministic Big Rock
        else:
            # 50% chance for a top 10% rock, 50% random
            if random.random() < 0.5:
                 start_txn = random.choice(sorted_txns_by_len[:max(1, len(sorted_txns_by_len)//10)])
            else:
                 start_txn = random.randint(0, workload.num_txns - 1)

        txn_seq = [start_txn]
        remaining_txns = set(range(workload.num_txns))
        remaining_txns.remove(start_txn)

        while remaining_txns:
            candidates = set()

            if i == 0:
                # Full scan for first sequence
                candidates = remaining_txns
            else:
                # Dynamic Big Rocks Heuristic
                # Add top available Big Rocks
                rocks_added = 0
                for t in sorted_txns_by_len:
                    if t in remaining_txns:
                        candidates.add(t)
                        rocks_added += 1
                        if rocks_added >= 5:
                            break

                # Add Random Samples
                if len(remaining_txns) > 20:
                    candidates.update(random.sample(list(remaining_txns), 15))
                else:
                    candidates.update(remaining_txns)

            # Calculate dynamic alpha based on sequence progress
            progress = len(txn_seq) / workload.num_txns
            alpha = start_alpha + (end_alpha - start_alpha) * progress

            # Weighted Selection
            best_t = -1
            best_score = float('inf')

            for t in candidates:
                cost = workload.get_opt_seq_cost(txn_seq + [t])
                # Score = Cost - (Alpha * Duration)
                score = cost - (alpha * txn_durations[t])

                if score < best_score:
                    best_score = score
                    best_t = t
                elif score == best_score:
                    if txn_durations[t] > txn_durations.get(best_t, 0):
                        best_t = t

            txn_seq.append(best_t)
            remaining_txns.remove(best_t)

        total_cost = workload.get_opt_seq_cost(txn_seq)
        candidates_pool.append((total_cost, txn_seq))

    # 3. Selection (Top 3 Unique)
    candidates_pool.sort(key=lambda x: x[0])
    unique_candidates = []
    seen_hashes = set()
    for cost, seq in candidates_pool:
        h = tuple(seq)
        if h not in seen_hashes:
            unique_candidates.append((cost, list(seq)))
            seen_hashes.add(h)
        if len(unique_candidates) >= 3:
            break

    if not unique_candidates and candidates_pool:
        unique_candidates = [(candidates_pool[0][0], list(candidates_pool[0][1]))]

    # Helper: LAHC with Multi-Scale Mutation
    def run_lahc(schedule, start_cost, budget):
        current_sched = list(schedule)
        current_cost = start_cost
        best_sched = list(schedule)
        best_cost = start_cost

        # History Configuration
        history_len = 50
        history = [start_cost] * history_len

        # Cooling: Last 20% of budget is strict Hill Climbing
        cooling_start = int(budget * 0.8)

        for k in range(budget):
            op_rand = random.random()
            neighbor = list(current_sched)

            # Operators:
            # 50% Single Insert (Fine-tuning)
            # 25% Micro-Block Insert (Local dependency fix, size 2-4)
            # 5% Macro-Block Insert (Structural change, size 5-10)
            # 20% Swap (Exploration)

            if op_rand < 0.50:
                # Single Insert
                idx1 = random.randint(0, len(neighbor) - 1)
                idx2 = random.randint(0, len(neighbor) - 1)
                if idx1 != idx2:
                    item = neighbor.pop(idx1)
                    neighbor.insert(idx2, item)

            elif op_rand < 0.75:
                # Micro-Block Insert (2-4 items)
                bsize = random.randint(2, 4)
                if len(neighbor) > bsize:
                    start = random.randint(0, len(neighbor) - bsize)
                    block = neighbor[start : start + bsize]
                    del neighbor[start : start + bsize]
                    dest = random.randint(0, len(neighbor))
                    neighbor[dest : dest] = block
                else:
                    continue

            elif op_rand < 0.80:
                # Macro-Block Insert (5-10 items) - Rare structural shift
                bsize = random.randint(5, 10)
                if len(neighbor) > bsize:
                    start = random.randint(0, len(neighbor) - bsize)
                    block = neighbor[start : start + bsize]
                    del neighbor[start : start + bsize]
                    dest = random.randint(0, len(neighbor))
                    neighbor[dest : dest] = block
                else:
                    continue

            else:
                # Swap
                idx1, idx2 = random.sample(range(len(neighbor)), 2)
                neighbor[idx1], neighbor[idx2] = neighbor[idx2], neighbor[idx1]

            new_cost = workload.get_opt_seq_cost(neighbor)

            # Acceptance Logic
            if k >= cooling_start:
                # Strict Hill Climbing (Cooling Phase)
                if new_cost <= current_cost:
                    current_sched = neighbor
                    current_cost = new_cost
                    if current_cost < best_cost:
                        best_cost = current_cost
                        best_sched = list(current_sched)
            else:
                # Late Acceptance Hill Climbing
                v = k % history_len
                if new_cost <= current_cost or new_cost <= history[v]:
                    current_sched = neighbor
                    current_cost = new_cost
                    if current_cost < best_cost:
                        best_cost = current_cost
                        best_sched = list(current_sched)
                history[v] = current_cost

        return best_cost, best_sched

    # 4. Refinement Strategy

    # Sprint Phase: Optimize top candidates
    sprint_results = []
    for cost, seq in unique_candidates:
        c, s = run_lahc(seq, cost, 300)
        sprint_results.append((c, s))

    sprint_results.sort(key=lambda x: x[0])
    winner_cost, winner_seq = sprint_results[0]

    # Marathon Phase: Deep optimization on best candidate
    final_cost, final_seq = run_lahc(winner_seq, winner_cost, 2500)

    return final_cost, final_seq
