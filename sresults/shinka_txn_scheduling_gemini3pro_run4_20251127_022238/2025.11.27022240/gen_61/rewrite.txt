def get_best_schedule(workload, num_seqs):
    """
    Hybrid scheduler combining strict cost evaluation and Big Rocks strategy
    with LAHC-based Funnel refinement.

    Strategies:
    1. Greedy: Use lexicographical selection (Cost, -Duration) from Code 2.
    2. Diversity: Randomize number of Big Rocks and Sample Size.
    3. Refinement: Funnel top candidates through Sprint and Marathon LAHC stages.
    """
    # 1. Pre-calculation
    # Compute transaction durations for heuristic guidance
    txn_durations = {t: workload.get_opt_seq_cost([t]) for t in range(workload.num_txns)}
    # Sort transactions by duration (descending) for Big Rocks lookup
    sorted_txns_by_len = sorted(range(workload.num_txns), key=lambda t: txn_durations[t], reverse=True)

    candidates_pool = []

    # 2. Greedy Construction Phase
    for i in range(num_seqs):
        # Random starting point for diversity
        start_txn = random.randint(0, workload.num_txns - 1)
        txn_seq = [start_txn]
        
        remaining_txns = set(range(workload.num_txns))
        remaining_txns.remove(start_txn)
        
        # Diversity parameters:
        # Vary the number of rocks and sample size slightly per iteration
        num_rocks = random.randint(3, 6)
        sample_size = random.randint(15, 25)

        while remaining_txns:
            candidates = set()
            
            if i == 0:
                # Baseline: Full scan for the first sequence
                candidates = remaining_txns
            else:
                # Hybrid Sampling: Big Rocks + Random
                
                # Big Rocks: Add top k longest available transactions
                rocks_added = 0
                for t in sorted_txns_by_len:
                    if t in remaining_txns:
                        candidates.add(t)
                        rocks_added += 1
                        if rocks_added >= num_rocks:
                            break
                
                # Random Sampling to fill the pool
                remaining_list = list(remaining_txns)
                if len(remaining_list) <= sample_size:
                    candidates.update(remaining_list)
                else:
                    # Fill until we reach sample_size
                    # We use a while loop with random.choice as simpler set filling
                    while len(candidates) < sample_size:
                         candidates.add(random.choice(remaining_list))

            # Selection: Strict Lexicographical (Minimize Cost, Maximize Duration)
            # This replaces the weighted alpha approach. 
            # We want the schedule with the lowest makespan. 
            # If tie, we pick the one with the longest duration (heaviest task).
            best_t = -1
            best_tuple = (float('inf'), float('-inf'))

            for t in candidates:
                cost = workload.get_opt_seq_cost(txn_seq + [t])
                
                # Tuple comparison: (Primary: Cost, Secondary: -Duration)
                # Lower cost is better. More negative (-duration) is better (i.e., larger duration).
                current_tuple = (cost, -txn_durations[t])
                
                if current_tuple < best_tuple:
                    best_tuple = current_tuple
                    best_t = t
            
            txn_seq.append(best_t)
            remaining_txns.remove(best_t)

        total_cost = workload.get_opt_seq_cost(txn_seq)
        candidates_pool.append((total_cost, txn_seq))

    # 3. Refinement Phase (Funnel Strategy)

    # Deduplicate candidates based on sequence hash
    unique_candidates_map = {}
    for cost, seq in candidates_pool:
        h = tuple(seq)
        if h not in unique_candidates_map or cost < unique_candidates_map[h][0]:
            unique_candidates_map[h] = (cost, seq)
    
    unique_candidates = list(unique_candidates_map.values())
    unique_candidates.sort(key=lambda x: x[0])
    
    # Take Top 3 candidates for the "Sprint"
    top_candidates = unique_candidates[:3]
    if not top_candidates:
        top_candidates = candidates_pool[:1]

    # Helper: LAHC with Enhanced Operators
    def run_lahc(schedule, start_cost, budget, use_kick=False):
        current_sched = list(schedule)
        current_cost = start_cost
        best_sched = list(schedule)
        best_cost = start_cost

        # LAHC History Buffer
        history_len = 50
        history = [start_cost] * history_len
        last_imp_idx = 0

        for k in range(budget):
            # Kick Mechanism: Shuffle segment if stagnant
            if use_kick and (k - last_imp_idx > 400):
                if len(current_sched) > 20:
                    idx = random.randint(0, len(current_sched) - 20)
                    seg_len = random.randint(10, 20)
                    end = idx + seg_len
                    segment = current_sched[idx:end]
                    random.shuffle(segment)
                    current_sched[idx:end] = segment
                    
                    current_cost = workload.get_opt_seq_cost(current_sched)
                    history = [current_cost] * history_len # Reset history
                    last_imp_idx = k
                    continue

            neighbor = list(current_sched)
            op_rand = random.random()
            
            # Operators: 
            # 60% Single Insert (Fine tuning, heavily favored in effective solutions)
            # 30% Block Insert (Moving dependency chains)
            # 10% Swap (Exploration)
            
            if op_rand < 0.60:
                # Single Insert
                idx1 = random.randint(0, len(neighbor) - 1)
                idx2 = random.randint(0, len(neighbor) - 1)
                if idx1 != idx2:
                    item = neighbor.pop(idx1)
                    neighbor.insert(idx2, item)
            
            elif op_rand < 0.90:
                # Block Insert
                bsize = random.randint(2, 6)
                if len(neighbor) > bsize:
                    start = random.randint(0, len(neighbor) - bsize)
                    block = neighbor[start : start + bsize]
                    del neighbor[start : start + bsize]
                    dest = random.randint(0, len(neighbor))
                    neighbor[dest:dest] = block
                else:
                    continue
            
            else:
                # Swap
                idx1, idx2 = random.sample(range(len(neighbor)), 2)
                neighbor[idx1], neighbor[idx2] = neighbor[idx2], neighbor[idx1]

            new_cost = workload.get_opt_seq_cost(neighbor)

            # LAHC Acceptance Logic
            v = k % history_len
            if new_cost <= current_cost or new_cost <= history[v]:
                current_sched = neighbor
                current_cost = new_cost
                if current_cost < best_cost:
                    best_cost = current_cost
                    best_sched = list(current_sched)
                    last_imp_idx = k
            
            history[v] = current_cost

        return best_cost, best_sched

    # Stage 1: Sprint (Top 3 -> 200 iters)
    # Quick optimization to distinguish best potential
    sprint_results = []
    for cost, seq in top_candidates:
        c, s = run_lahc(seq, cost, 200, use_kick=False)
        sprint_results.append((c, s))
    
    sprint_results.sort(key=lambda x: x[0])
    
    # Stage 2: Marathon (Winner -> 2500 iters)
    # Deep optimization with kicks enabled
    winner_cost, winner_seq = sprint_results[0]
    final_cost, final_seq = run_lahc(winner_seq, winner_cost, 2500, use_kick=True)

    return final_cost, final_seq


def get_random_costs():
    """
    Evaluate scheduling algorithm on three different workloads.

    Returns:
        Tuple of (total_makespan, list_of_schedules, execution_time)
    """
    start_time = time.time()
    workload_size = 100

    # Workload 1: Complex mixed read/write transactions
    workload = Workload(WORKLOAD_1)
    makespan1, schedule1 = get_best_schedule(workload, 10)
    cost1 = workload.get_opt_seq_cost(schedule1)

    # Workload 2: Simple read-then-write pattern
    workload2 = Workload(WORKLOAD_2)
    makespan2, schedule2 = get_best_schedule(workload2, 10)
    cost2 = workload2.get_opt_seq_cost(schedule2)

    # Workload 3: Minimal read/write operations
    workload3 = Workload(WORKLOAD_3)
    makespan3, schedule3 = get_best_schedule(workload3, 10)
    cost3 = workload3.get_opt_seq_cost(schedule3)

    total_makespan = cost1 + cost2 + cost3
    schedules = [schedule1, schedule2, schedule3]
    execution_time = time.time() - start_time

    return total_makespan, schedules, execution_time
