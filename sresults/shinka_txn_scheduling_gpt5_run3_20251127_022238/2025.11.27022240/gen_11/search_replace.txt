<DESCRIPTION>
I replace the simple per-restart beam search with a stronger, time-bounded beam that uses:
- Shared caches across all steps/restarts: a prefix cost cache and an extension-cost (Î”) cache to avoid recomputing workload.get_opt_seq_cost on the same prefixes repeatedly.
- Branch ordering by marginal cost: at each expansion, we rank candidates by delta = cost(seq+[cand]) - cost(seq) and only expand the cheapest ones.
- Shallow two-step lookahead for ranking: for each expanded prefix, we sample a few next-next candidates and use the best two-step cost to sort the beam, reducing myopia while keeping branching controlled.
- Incumbent-based pruning and greedy completion: maintain the best full sequence found so far and prune prefixes whose cost already exceeds it; periodically greedily complete the current best prefix to refresh the incumbent early.
- Lightweight local search: after construction, perform adjacent swaps and sampled insertions to refine the final sequence using the shared cache.

This combination targets makespan directly with cost-aware decisions and improves both exploration and exploitation under a tight time budget, which should yield better schedules than pure heuristic or naive beam approaches.

def get_best_schedule(workload, num_seqs):

    Find a near-optimal schedule using beam search with memoized partial costs
    and multiple random restarts.

    Args:
        workload: Workload object containing transaction data
        num_seqs: Number of random restarts for the beam search

    Returns:
        Tuple of (lowest makespan, corresponding schedule)
    """

        N = workload.num_txns
        # Dynamically size the beam and branching according to problem size
        beam_width = min(max(4, N // 8), 32)
        branch_factor = min(max(6, N // 10), 24)

        all_txns = list(range(N))
        # Memoize costs for partial prefixes to avoid recomputation
        cost_cache = {}

        def eval_seq_cost(seq):
            key = tuple(seq)
            cached = cost_cache.get(key)
            if cached is not None:
                return cached
            c = workload.get_opt_seq_cost(seq)
            cost_cache[key] = c
            return c

        # Initialize beam with the best singletons from a random pool
        init_pool_size = min(len(all_txns), beam_width * 2)
        init_candidates = random.sample(all_txns, init_pool_size) if init_pool_size > 0 else all_txns[:]
        beam = []
        for t in init_candidates:
            seq = [t]
            rem = set(all_txns)
            rem.remove(t)
            cost = eval_seq_cost(seq)
            beam.append((cost, seq, rem))
        beam.sort(key=lambda x: x[0])
        beam = beam[:max(1, min(beam_width, len(beam)))]

        # Expand the beam until full sequences are built
        steps = N - 1
        for _ in range(steps):
            new_beam = []
            for cost, seq, rem in beam:
                if not rem:
                    # Already complete; keep as-is
                    new_beam.append((cost, seq, rem))
                    continue

                rem_list = list(rem)

                if len(rem_list) <= branch_factor:
                    candidates = rem_list
                else:
                    candidates = random.sample(rem_list, branch_factor)

                    new_seq = seq + [cand]
                    new_rem = rem.copy()
                    new_rem.remove(cand)
                    new_cost = eval_seq_cost(new_seq)
                    new_beam.append((new_cost, new_seq, new_rem))

                break

            new_beam.sort(key=lambda x: x[0])

            seen = set()

                key = tuple(entry[1])
                if key in seen:
                    continue
                seen.add(key)

                if len(unique) >= beam_width:
                    break
            beam = unique

        # Return the best full sequence from the beam
        best_entry = min(beam, key=lambda x: x[0])
        return best_entry[0], best_entry[1]

    best_seq = None

    # Multiple random restarts for robustness
    restarts = max(1, int(num_seqs))
    for _ in range(restarts):
        cost, seq = run_beam_search()
        if cost < best_cost:

def get_best_schedule(workload, num_seqs):
    """
    Beam search with marginal-cost ordering, shallow lookahead, shared caches,
    incumbent-based pruning, and a lightweight local refinement step.
    """

    N = workload.num_txns
    start_time = time.time()
    # Per-workload time budget (seconds)
    time_budget_sec = 0.45

    def time_left():
        return (time.time() - start_time) < time_budget_sec

    # Shared caches across all phases/restarts
    cost_cache = {}
    delta_cache = {}

    def eval_cost(seq):
        key = tuple(seq)
        cached = cost_cache.get(key)
        if cached is not None:
            return cached
        c = workload.get_opt_seq_cost(seq)
        cost_cache[key] = c
        return c

    def eval_ext_cost(prefix_tuple, cand):
        key = (prefix_tuple, cand)
        cached = delta_cache.get(key)
        if cached is not None:
            return cached
        seq = list(prefix_tuple) + [cand]
        c = eval_cost(seq)
        delta_cache[key] = c
        return c

    # Precompute singleton costs to warm the cache and seed initial beam
    singleton_cost = {}
    for t in range(N):
        if not time_left():
            break
        c = eval_cost([t])
        singleton_cost[t] = c

    # Beam and branching settings
    beam_width = min(36, max(8, N // 6))
    branch_factor = min(24, max(8, N // 8))
    lookahead_k = 4

    # Incumbent best complete sequence for pruning
    incumbent_cost = float('inf')
    incumbent_seq = None

    def greedy_finish(seq, rem_set):
        seq_out = list(seq)
        rem = set(rem_set)
        cur_cost = eval_cost(seq_out)
        while rem and time_left():
            rem_list = list(rem)
            k = min(branch_factor, len(rem_list))
            if len(rem_list) > k:
                cand_pool = random.sample(rem_list, k)
            else:
                cand_pool = rem_list
            best_t = None
            best_c = float('inf')
            prefix_tuple = tuple(seq_out)
            for t in cand_pool:
                c = eval_ext_cost(prefix_tuple, t)
                if c < best_c:
                    best_c = c
                    best_t = t
            if best_t is None:
                best_t = rem_list[0]
                best_c = eval_ext_cost(prefix_tuple, best_t)
            seq_out.append(best_t)
            rem.remove(best_t)
            cur_cost = best_c
            if cur_cost >= incumbent_cost:
                break
        return cur_cost, seq_out

    def run_beam_search():
        nonlocal incumbent_cost, incumbent_seq
        all_txns = list(range(N))

        # Initialize beam with top-k singletons by cost
        singletons = [(singleton_cost.get(t, eval_cost([t])), [t], set(x for x in all_txns if x != t)) for t in all_txns]
        singletons.sort(key=lambda x: x[0])
        beam = singletons[:max(1, min(beam_width, len(singletons)))]

        remaining_steps = N - 1
        for _ in range(remaining_steps):
            if not time_left():
                break
            new_beam = []
            for cost_so_far, seq, rem in beam:
                if not rem:
                    new_beam.append((cost_so_far, seq, rem))
                    if cost_so_far < incumbent_cost:
                        incumbent_cost, incumbent_seq = cost_so_far, seq[:]
                    continue
                if cost_so_far >= incumbent_cost:
                    continue

                prefix_tuple = tuple(seq)
                rem_list = list(rem)

                # Order candidates by marginal cost (delta)
                if len(rem_list) > branch_factor * 3:
                    cand_pool = random.sample(rem_list, branch_factor * 3)
                else:
                    cand_pool = rem_list

                scored = []
                for cand in cand_pool:
                    if not time_left():
                        break
                    ext_cost = eval_ext_cost(prefix_tuple, cand)
                    delta = ext_cost - cost_so_far
                    scored.append((delta, ext_cost, cand))
                if not scored:
                    continue

                scored.sort(key=lambda x: x[0])
                top_cands = scored[:min(branch_factor, len(scored))]

                for _, ext_cost, cand in top_cands:
                    if not time_left():
                        break
                    new_seq = seq + [cand]
                    new_rem = rem.copy()
                    new_rem.remove(cand)

                    # Shallow lookahead using a small sample of next-next transactions
                    la_score = ext_cost
                    if new_rem:
                        la_pool = list(new_rem)
                        la_sample = la_pool if len(la_pool) <= lookahead_k else random.sample(la_pool, lookahead_k)
                        best_la = float('inf')
                        new_prefix_tuple = tuple(new_seq)
                        for nxt in la_sample:
                            c2 = eval_ext_cost(new_prefix_tuple, nxt)
                            if c2 < best_la:
                                best_la = c2
                        la_score = best_la

                    new_beam.append((ext_cost, new_seq, new_rem, la_score))

            if not new_beam:
                break

            # Keep top beam by lookahead score while preserving the actual prefix cost
            new_beam.sort(key=lambda x: x[3])
            unique = []
            seen = set()
            for entry in new_beam:
                key = tuple(entry[1])
                if key in seen:
                    continue
                seen.add(key)
                unique.append((entry[0], entry[1], entry[2]))
                if len(unique) >= beam_width:
                    break
            beam = unique

            # Greedily complete the best prefix periodically to tighten pruning
            if beam and time_left():
                best_prefix = min(beam, key=lambda x: x[0])
                if best_prefix[0] < incumbent_cost:
                    c_try, seq_try = greedy_finish(best_prefix[1], best_prefix[2])
                    if len(seq_try) == N and c_try < incumbent_cost:
                        incumbent_cost, incumbent_seq = c_try, seq_try

        # Finish all prefixes greedily; pick best
        best_cost = incumbent_cost
        best_seq_local = incumbent_seq
        for cost_so_far, seq, rem in beam:
            if not time_left():
                break
            c, s = greedy_finish(seq, rem)
            if len(s) == N and c < best_cost:
                best_cost, best_seq_local = c, s

        if best_seq_local is None:
            seq = all_txns[:]
            random.shuffle(seq)
            best_seq_local = seq
            best_cost = eval_cost(seq)
        return best_cost, best_seq_local

    def local_improve(seq, current_cost):
        best_seq = seq[:]
        best_cost = current_cost

        # One pass of adjacent swaps
        for i in range(len(best_seq) - 1):
            if not time_left():
                break
            cand = best_seq[:]
            cand[i], cand[i + 1] = cand[i + 1], cand[i]
            c = eval_cost(cand)
            if c < best_cost:
                best_cost = c
                best_seq = cand

        # Limited random insertions
        trials = 40
        while trials > 0 and time_left():
            trials -= 1
            i = random.randrange(len(best_seq))
            j = random.randrange(len(best_seq))
            if i == j:
                continue
            cand = best_seq[:]
            val = cand.pop(i)
            cand.insert(j, val)
            c = eval_cost(cand)
            if c < best_cost:
                best_cost = c
                best_seq = cand

        return best_cost, best_seq

    global_best_cost = float('inf')
    global_best_seq = None

    # Time-bounded restarts
    restart = 0
    while time_left() and restart < max(1, int(num_seqs)):
        cost, seq = run_beam_search()
        if time_left():
            cost, seq = local_improve(seq, cost)
        if cost < global_best_cost:
            global_best_cost, global_best_seq = cost, seq
        restart += 1

    if global_best_seq is None:
        global_best_seq = list(range(N))
        random.shuffle(global_best_seq)
        global_best_cost = eval_cost(global_best_seq)

    return global_best_cost, global_best_seq
