def get_best_schedule(workload, num_seqs):
    """
    Get optimal schedule using a Genetic Algorithm to minimize makespan.
    
    Args:
        workload: Workload object containing transaction data
        num_seqs: Used to scale the population size/effort (heuristic)

    Returns:
        Tuple of (lowest makespan, corresponding schedule)
    """
    # GA Parameters
    # Scale population slightly with num_seqs, but keep bounds to ensure performance
    POPULATION_SIZE = max(30, num_seqs * 2)
    GENERATIONS = 25
    ELITISM_COUNT = 2
    MUTATION_RATE = 0.3
    TOURNAMENT_SIZE = 4
    
    num_txns = workload.num_txns
    
    # Memoization cache for the cost function to speed up re-evaluations
    cost_cache = {}

    def get_fitness(schedule_seq):
        # Convert list to tuple for dictionary key
        seq_tuple = tuple(schedule_seq)
        if seq_tuple in cost_cache:
            return cost_cache[seq_tuple]
        
        cost = workload.get_opt_seq_cost(schedule_seq)
        cost_cache[seq_tuple] = cost
        return cost

    # Initialize population with random permutations
    population = []
    base_seq = list(range(num_txns))
    for _ in range(POPULATION_SIZE):
        p = base_seq[:]
        random.shuffle(p)
        population.append(p)
        
    best_schedule = None
    min_makespan = float('inf')

    # Evolution Loop
    for generation in range(GENERATIONS):
        # Evaluate fitness
        pop_with_fitness = []
        for ind in population:
            f = get_fitness(ind)
            pop_with_fitness.append((ind, f))
            if f < min_makespan:
                min_makespan = f
                best_schedule = ind[:]
        
        # Sort by fitness (makespan -> lower is better)
        pop_with_fitness.sort(key=lambda x: x[1])
        
        # Elitism: carry over best individuals
        new_population = [x[0] for x in pop_with_fitness[:ELITISM_COUNT]]
        
        # Create offspring
        while len(new_population) < POPULATION_SIZE:
            # Tournament Selection
            candidates = random.sample(pop_with_fitness, TOURNAMENT_SIZE)
            candidates.sort(key=lambda x: x[1])
            parent1 = candidates[0][0]
            parent2 = candidates[1][0]
            
            # Order Crossover (OX1)
            # Select random segment
            start, end = sorted(random.sample(range(num_txns), 2))
            
            child = [-1] * num_txns
            # Copy segment from P1
            child[start:end] = parent1[start:end]
            
            # Fill remaining from P2 in order
            current_pos = end
            for item in parent2:
                if item not in child[start:end]:
                    if current_pos >= num_txns:
                        current_pos = 0
                    child[current_pos] = item
                    current_pos += 1
            
            # Shift Mutation (move a transaction to a different priority slot)
            if random.random() < MUTATION_RATE:
                idx_from, idx_to = random.sample(range(num_txns), 2)
                val = child.pop(idx_from)
                child.insert(idx_to, val)
                
            new_population.append(child)
            
        population = new_population

    # Final evaluation to ensure we return the absolute best found
    final_cost = get_fitness(best_schedule)
    return final_cost, best_schedule


def get_random_costs():
    """
    Evaluate scheduling algorithm on three different workloads.
    
    Returns:
        Tuple of (total_makespan, list_of_schedules, execution_time)
    """
    start_time = time.time()
    
    # Workload 1: Complex mixed read/write transactions
    workload1 = Workload(WORKLOAD_1)
    makespan1, schedule1 = get_best_schedule(workload1, 10)

    # Workload 2: Simple read-then-write pattern
    workload2 = Workload(WORKLOAD_2)
    makespan2, schedule2 = get_best_schedule(workload2, 10)

    # Workload 3: Minimal read/write operations
    workload3 = Workload(WORKLOAD_3)
    makespan3, schedule3 = get_best_schedule(workload3, 10)
    
    total_makespan = makespan1 + makespan2 + makespan3
    schedules = [schedule1, schedule2, schedule3]
    execution_time = time.time() - start_time
    
    return total_makespan, schedules, execution_time
