def get_best_schedule(workload, num_seqs):
    """
    Get optimal schedule using Average-Lookahead Beam Search and Chain-Ruin ILS.

    Innovations:
    1. Average-Lookahead: Evaluates candidates based on how well they play with the 
       top-4 most urgent remaining items (mix of min and avg cost).
    2. Chain-Ruin ILS: Removes contiguous blocks of transactions to disrupt 
       sub-optimal dependency chains, then re-inserts them.
    3. Balanced Urgency: 0.4 Duration / 0.6 Conflict Vol.

    Args:
        workload: Workload object
        num_seqs: Budget parameter

    Returns:
        (lowest_makespan, schedule)
    """

    num_txns = workload.num_txns

    # --- 1. METRIC PRECOMPUTATION ---
    txn_durations = {}
    txn_rw_sets = {}

    for t in range(num_txns):
        # Duration
        try:
            d = workload.txns[t][0][3]
        except:
            d = 1.0
        txn_durations[t] = d

        # R/W Sets
        reads = set()
        writes = set()
        try:
            ops_str = workload.txns[t][0][1]
            if isinstance(ops_str, str):
                for op in ops_str.split():
                    if '-' in op:
                        parts = op.split('-')
                        if len(parts) == 2:
                            op_type, key = parts
                            if op_type == 'r': reads.add(key)
                            elif op_type == 'w': writes.add(key)
        except:
            pass
        txn_rw_sets[t] = (reads, writes)

    # Compute Conflict Volume
    txn_conflict_vol = {t: 0.0 for t in range(num_txns)}

    if num_txns < 2500:
        for i in range(num_txns):
            r1, w1 = txn_rw_sets[i]
            vol = 0.0
            for j in range(num_txns):
                if i == j: continue
                r2, w2 = txn_rw_sets[j]
                if not w1.isdisjoint(w2) or not w1.isdisjoint(r2) or not r1.isdisjoint(w2):
                    vol += txn_durations[j]
            txn_conflict_vol[i] = vol

    # Urgency Score
    max_vol = max(txn_conflict_vol.values()) if txn_conflict_vol else 1.0
    if max_vol == 0: max_vol = 1.0
    max_dur = max(txn_durations.values()) if txn_durations else 1.0

    txn_urgency = {}
    for t in range(num_txns):
        # Slightly favor Conflict Volume (0.6) to prioritize unlocking dependencies
        u = 0.4 * (txn_durations[t] / max_dur) + 0.6 * (txn_conflict_vol[t] / max_vol)
        txn_urgency[t] = u

    # --- 2. BEAM SEARCH WITH AVERAGE LOOKAHEAD ---

    BEAM_WIDTH = max(5, int(num_seqs))

    # Beam State: (immediate_cost, schedule_list, remaining_list)
    beam = [(0, [], list(range(num_txns)))]

    for _ in range(num_txns):
        candidates_pool = []

        for p_cost, p_sched, p_remain in beam:

            # Sort remaining by Urgency
            sorted_remain = sorted(p_remain, key=lambda x: txn_urgency[x], reverse=True)

            to_evaluate = set()

            # 1. Deterministic Top 2
            to_evaluate.update(sorted_remain[:2])

            # 2. Cubic Weighted Sampling from next 15
            # Broaden the pool slightly compared to previous gens to find better fits
            if len(sorted_remain) > 2:
                pool = sorted_remain[2 : 17] # Top 15 after first 2
                weights = [txn_urgency[x]**3 for x in pool]
                
                k_samples = min(3, len(pool))
                if k_samples > 0:
                    samples = random.choices(pool, weights=weights, k=k_samples)
                    to_evaluate.update(samples)

            # 3. Exploration (1 random)
            if len(sorted_remain) > 20:
                to_evaluate.update(random.sample(sorted_remain, 1))

            # Evaluation Loop
            for cand in to_evaluate:
                curr_sched = p_sched + [cand]
                curr_cost = workload.get_opt_seq_cost(curr_sched)

                # Lookahead: Evaluate with Top-4 Urgent Items
                metric_score = curr_cost
                
                probes = []
                count = 0
                for r in sorted_remain:
                    if r == cand: continue
                    probes.append(r)
                    count += 1
                    if count >= 4: break # Check top 4
                
                if probes:
                    probe_costs = []
                    for p in probes:
                        # Cost if we follow cand with p
                        probe_costs.append(workload.get_opt_seq_cost(curr_sched + [p]))
                    
                    if probe_costs:
                        min_c = min(probe_costs)
                        avg_c = sum(probe_costs) / len(probe_costs)
                        # Metric: Balance between Best Fit (min) and General Compatibility (avg)
                        # 0.6 Min + 0.4 Avg
                        metric_score = 0.6 * min_c + 0.4 * avg_c

                # Score Tuple: (LookaheadMetric, ImmediateCost, -Urgency)
                score = (metric_score, curr_cost, -txn_urgency[cand])
                
                rem_c = list(p_remain)
                rem_c.remove(cand)
                
                candidates_pool.append((score, curr_sched, rem_c))

        # Pruning
        candidates_pool.sort(key=lambda x: x[0])
        beam = [(x[0][1], x[1], x[2]) for x in candidates_pool[:BEAM_WIDTH]]

    best_state = beam[0]
    current_cost = best_state[0]
    current_schedule = best_state[1]

    # --- 3. CHAIN-RUIN ILS ---

    def local_descent(sched, start_c, passes=1, window=10):
        """Windowed insertion to polish schedule."""
        s = list(sched)
        c = start_c
        
        for _ in range(passes):
            improved = False
            indices = list(range(len(s)))
            random.shuffle(indices)
            
            for i in indices:
                item = s[i]
                temp = s[:i] + s[i+1:]
                
                start = max(0, i - window)
                end = min(len(temp), i + window)
                
                best_p = -1
                best_v = c
                
                for p in range(start, end + 1):
                    cand = temp[:p] + [item] + temp[p:]
                    # Simple optimization: skip if same position (approx)
                    if p == i: continue
                    
                    val = workload.get_opt_seq_cost(cand)
                    if val < best_v:
                        best_v = val
                        best_p = p
                
                if best_p != -1:
                    s = temp[:best_p] + [item] + temp[best_p:]
                    c = best_v
                    improved = True
            
            if not improved: break
        return c, s

    # Initial Polish
    current_cost, current_schedule = local_descent(current_schedule, current_cost, passes=2)

    # ILS Loop
    max_kicks = 5
    
    for _ in range(max_kicks):
        neighbor = list(current_schedule)
        
        # Strategy Selection
        r = random.random()
        
        # 1. Chain Ruin (Contiguous Block Removal)
        # Removes a block, which might be a dependency chain, and lets re-insertion find better spots
        if r < 0.5:
            if len(neighbor) > 10:
                block_size = random.randint(4, 10)
                start_idx = random.randint(0, len(neighbor) - block_size)
                
                # Extract block
                removed_items = neighbor[start_idx : start_idx + block_size]
                del neighbor[start_idx : start_idx + block_size]
                
                # Re-insert: Sort by Urgency or maintain relative order?
                # Maintaining relative order preserves the chain logic, 
                # but sorting by urgency might fix priority issues.
                # Let's try: Re-insert highest urgency first.
                removed_items.sort(key=lambda x: txn_urgency[x], reverse=True)
                
                for item in removed_items:
                    best_p = -1
                    best_v = float('inf')
                    
                    # Coarse search
                    step = 2 if len(neighbor) > 100 else 1
                    for p in range(0, len(neighbor) + 1, step):
                        cand = neighbor[:p] + [item] + neighbor[p:]
                        val = workload.get_opt_seq_cost(cand)
                        if val < best_v:
                            best_v = val
                            best_p = p
                    
                    # Refine
                    if step > 1 and best_p != -1:
                        for p in range(max(0, best_p-2), min(len(neighbor), best_p+3)):
                            cand = neighbor[:p] + [item] + neighbor[p:]
                            val = workload.get_opt_seq_cost(cand)
                            if val < best_v:
                                best_v = val
                                best_p = p
                                
                    if best_p != -1:
                        neighbor.insert(best_p, item)
                    else:
                        neighbor.append(item)
        
        # 2. Multi-Swap (Random Perturbation)
        else:
            for _ in range(4):
                i, j = random.randint(0, num_txns-1), random.randint(0, num_txns-1)
                neighbor[i], neighbor[j] = neighbor[j], neighbor[i]

        # Repair
        kick_cost = workload.get_opt_seq_cost(neighbor)
        new_cost, new_sched = local_descent(neighbor, kick_cost, passes=1)
        
        # Acceptance
        if new_cost < current_cost:
            current_cost = new_cost
            current_schedule = new_sched

    return current_cost, current_schedule


def get_random_costs():
    """Evaluate scheduling algorithm on three different workloads."""
    start_time = time.time()
    
    num_seqs = 10
    
    workload1 = Workload(WORKLOAD_1)
    makespan1, schedule1 = get_best_schedule(workload1, num_seqs)

    workload2 = Workload(WORKLOAD_2)
    makespan2, schedule2 = get_best_schedule(workload2, num_seqs)

    workload3 = Workload(WORKLOAD_3)
    makespan3, schedule3 = get_best_schedule(workload3, num_seqs)
    
    total_makespan = makespan1 + makespan2 + makespan3
    schedules = [schedule1, schedule2, schedule3]
    execution_time = time.time() - start_time
    
    return total_makespan, schedules, execution_time
