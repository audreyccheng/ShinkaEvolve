def get_best_schedule(workload, num_seqs):
    """
    Get optimal schedule using Cubic-Weighted Beam Search and Stagnation-Based Adaptive ILS.

    Innovations:
    1. Tuned Urgency: Balances Duration and Conflict Volume to identify bottleneck transactions.
    2. Multi-Target Lookahead Beam Search: Evaluates candidates by seeing if they allow *any* of
       the top-2 urgent items to follow efficiently.
    3. Adaptive ILS: Increases perturbation intensity (Swap -> Block Move -> Ruin & Recreate)
       when the search stagnates, ensuring robust escape from local optima.

    Args:
        workload: Workload object
        num_seqs: Budget parameter

    Returns:
        (lowest_makespan, schedule)
    """

    num_txns = workload.num_txns

    # --- 1. METRIC PRECOMPUTATION ---
    txn_durations = {}
    txn_rw_sets = {}

    for t in range(num_txns):
        # Duration
        try:
            d = workload.txns[t][0][3]
        except:
            d = 1.0
        txn_durations[t] = d

        # R/W Sets
        reads = set()
        writes = set()
        try:
            ops_str = workload.txns[t][0][1]
            if isinstance(ops_str, str):
                for op in ops_str.split():
                    if '-' in op:
                        parts = op.split('-')
                        if len(parts) == 2:
                            op_type, key = parts
                            if op_type == 'r': reads.add(key)
                            elif op_type == 'w': writes.add(key)
        except:
            pass
        txn_rw_sets[t] = (reads, writes)

    # Compute Conflict Volume
    # Sum of durations of all transactions that conflict with T
    txn_conflict_vol = {t: 0.0 for t in range(num_txns)}

    if num_txns < 2000:
        for i in range(num_txns):
            r1, w1 = txn_rw_sets[i]
            vol = 0.0
            for j in range(num_txns):
                if i == j: continue
                r2, w2 = txn_rw_sets[j]
                # Conflict: (W1 n (W2 u R2)) or (R1 n W2)
                if not w1.isdisjoint(w2) or not w1.isdisjoint(r2) or not r1.isdisjoint(w2):
                    vol += txn_durations[j]
            txn_conflict_vol[i] = vol

    # Urgency Score
    max_vol = max(txn_conflict_vol.values()) if txn_conflict_vol else 1.0
    if max_vol == 0: max_vol = 1.0
    max_dur = max(txn_durations.values()) if txn_durations else 1.0

    txn_urgency = {}
    for t in range(num_txns):
        # Weighting: Duration is primary, but Conflict Volume is a strong secondary factor.
        # Ratio 1.0 : 0.8
        u = (txn_durations[t] / max_dur) + 0.8 * (txn_conflict_vol[t] / max_vol)
        txn_urgency[t] = u

    # --- 2. BEAM SEARCH WITH LOOKAHEAD ---

    BEAM_WIDTH = max(5, int(num_seqs))

    # Beam State: (cost, schedule_list, remaining_list)
    beam = [(0, [], list(range(num_txns)))]

    for _ in range(num_txns):
        candidates_pool = []

        for p_cost, p_sched, p_remain in beam:

            # Sort remaining by Urgency
            sorted_remain = sorted(p_remain, key=lambda x: txn_urgency[x], reverse=True)

            next_candidates = set()

            # 1. Deterministic Top 3 (Greedy)
            next_candidates.update(sorted_remain[:3])

            # 2. Cubic Weighted Sampling (Stochastic)
            if len(sorted_remain) > 3:
                pool = sorted_remain[3:]
                # Cubed weights to heavily favor top-end urgency while allowing some variance
                weights = [txn_urgency[x]**3 for x in pool]

                k_samples = min(4, len(pool))
                if k_samples > 0:
                    samples = random.choices(pool, weights=weights, k=k_samples)
                    next_candidates.update(samples)

            # 3. Random Diversity (Explore)
            if len(sorted_remain) > 10:
                next_candidates.update(random.sample(sorted_remain, 1))

            # Evaluation Loop
            for c in next_candidates:
                sched_c = p_sched + [c]
                cost_c = workload.get_opt_seq_cost(sched_c)

                # Lookahead: Check compatibility with Top-2 remaining urgent items
                metric_score = cost_c

                probes = []
                count = 0
                for r in sorted_remain:
                    if r == c: continue
                    probes.append(r)
                    count += 1
                    if count >= 2: break

                if probes:
                    # Calculate cost if we schedule a probe immediately after c
                    la_costs = []
                    for p in probes:
                        la_costs.append(workload.get_opt_seq_cost(sched_c + [p]))

                    if la_costs:
                        # Optimistic lookahead: can we schedule *at least one* important task efficiently?
                        metric_score = min(la_costs)

                # Metric Tuple: (LookaheadCost, ImmediateCost, -Urgency)
                # Primary: Minimize Lookahead Cost
                # Secondary: Minimize Immediate Cost
                # Tertiary: Maximize Urgency (break ties with heavy items)
                metric = (metric_score, cost_c, -txn_urgency[c])

                new_remain = list(p_remain)
                new_remain.remove(c)

                candidates_pool.append((metric, sched_c, new_remain))
