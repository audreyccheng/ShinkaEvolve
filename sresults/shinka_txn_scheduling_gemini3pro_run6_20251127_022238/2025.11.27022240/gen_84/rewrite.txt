    while current != os.path.dirname(current):  # Stop at filesystem root

        '/Users/audreycc/Documents/Work/LLMTxn/ADRS-Exps/ShinkaEvolve',  # Absolute path fallback for macOS
    ]

        candidate = os.path.join(root, 'openevolve_examples', 'txn_scheduling')
        if os.path.exists(candidate):
            return root

def get_best_schedule(workload, num_seqs):
    """
    Get optimal schedule using Quartic-Weighted Beam Search and Adaptive ILS.
    
    Algorithm Components:
    1. Beam Search:
       - Width: Expanded to improve global search capability.
       - Sampling: 5 Deterministic Anchors (Heavy) + Quartic Weighted Random (d^4).
       - Lookahead: Evaluates top-5 remaining heavy transactions to pick the path 
         that minimizes 'idle time' growth (Cost - TotalDuration).
    2. Adaptive ILS:
       - Ruin: Aggressively removes 6-10 transactions.
       - Recreate: Greedy insertion of removed items (heaviest first).
       - Refinement: Multi-stage local search (Insertions -> Block Moves -> Swaps).
       - Acceptance: Decaying threshold acceptance to allow exploration.

    Args:
        workload: Workload object
        num_seqs: Scaling factor for beam width
    """
    # --- Configuration ---
    # Beam Search
    BEAM_WIDTH = int(max(20, num_seqs * 3.0))
    SAMPLES_PER_NODE = 30
    DETERMINISTIC_ANCHORS = 5
    MAX_CHILDREN_PER_PARENT = 5
    
    # Lookahead
    LOOKAHEAD_TARGETS = 5
    
    # ILS
    ILS_CYCLES = 12
    RUIN_MIN, RUIN_MAX = 6, 10
    THRESHOLD_START = 0.015 # 1.5%

    num_txns = workload.num_txns

    # --- Precompute ---
    # Individual transaction costs (durations)
    txn_durations = {t: workload.get_opt_seq_cost([t]) for t in range(num_txns)}
    # Sorted by duration descending (LPT)
    sorted_lpt = sorted(range(num_txns), key=lambda t: txn_durations[t], reverse=True)

    # --- Beam Search ---
    
    # Initialization
    start_candidates = set()
    start_candidates.update(sorted_lpt[:BEAM_WIDTH])
    attempts = 0
    # Add some random starting points for diversity
    while len(start_candidates) < BEAM_WIDTH * 1.5 and attempts < num_txns:
        start_candidates.add(random.randint(0, num_txns - 1))
        attempts += 1
        
    beam = []
    for t in start_candidates:
        rem = set(range(num_txns))
        rem.remove(t)
        c = txn_durations[t]
        beam.append({
            'cost': c,
            'total_dur': c,
            'seq': [t],
            'rem': rem,
            'score': 0
        })
    
    # Sort by 'density' (Cost - TotalDuration)
    # Ideally 0 for the first item, but consistent scoring matters
    beam.sort(key=lambda x: (x['cost'] - x['total_dur'], -x['total_dur']))
    beam = beam[:BEAM_WIDTH]

    # Main Loop
    for _ in range(num_txns - 1):
        candidates = []
        
        for p_idx, parent in enumerate(beam):
            rem_list = list(parent['rem'])
            if not rem_list: continue

            # 1. Identify Samples
            samples = set()
            
            # A. Deterministic Anchors (Top-K LPT from remaining)
            # Efficiently find intersection of sorted_lpt and rem
            count = 0
            for t in sorted_lpt:
                if t in parent['rem']:
                    samples.add(t)
                    count += 1
                    if count >= DETERMINISTIC_ANCHORS: break
            
            # B. Weighted Random
            needed = SAMPLES_PER_NODE - len(samples)
            if needed > 0:
                pool = [x for x in rem_list if x not in samples]
                if pool:
                    if len(pool) <= needed:
                        samples.update(pool)
                    else:
                        # Quartic weights (d^4)
                        weights = [txn_durations[x]**4 for x in pool]
                        try:
                            # random.choices is efficient
                            chosen = random.choices(pool, weights=weights, k=needed * 2)
                            samples.update(chosen)
                        except ValueError:
                            pass
                        
                        # Fill gaps if duplicates occurred
                        if len(samples) < SAMPLES_PER_NODE:
                            rem_pool = [x for x in pool if x not in samples]
                            if rem_pool:
                                fill_k = min(len(rem_pool), SAMPLES_PER_NODE - len(samples))
                                samples.update(random.sample(rem_pool, fill_k))

            # 2. Evaluate Samples
            for t in samples:
                new_seq = parent['seq'] + [t]
                new_cost = workload.get_opt_seq_cost(new_seq)
                new_total_dur = parent['total_dur'] + txn_durations[t]
                
                # Base Score: (Makespan - Sum of Durations)
                # This proxies "Idle Time". We want to minimize this.
                base_score = new_cost - new_total_dur
                
                candidates.append({
                    'cost': new_cost,
                    'total_dur': new_total_dur,
                    'seq': new_seq,
                    'rem': parent['rem'], # Ref
                    'added': t,
                    'p_idx': p_idx,
                    'base_score': base_score
                })

        # --- Lookahead & Pruning ---
        
        # Pre-filter to reduce Lookahead overhead
        # Sort by base_score (asc) and total_dur (desc)
        candidates.sort(key=lambda x: (x['base_score'], -x['total_dur']))
        
        # Deduplicate states (set of remaining items)
        # Only keep the best path to any specific set of remaining items
        lookahead_candidates = []
        seen_states = set()
        # Limit the number of candidates entering lookahead to keep speed up
        # 2x Beam Width is usually sufficient
        la_pool_size = int(BEAM_WIDTH * 2.0)
        
        for cand in candidates:
            new_rem = cand['rem'].copy()
            new_rem.remove(cand['added'])
            state_key = frozenset(new_rem)
            
            if state_key not in seen_states:
                seen_states.add(state_key)
                cand['new_rem'] = new_rem
                lookahead_candidates.append(cand)
                if len(lookahead_candidates) >= la_pool_size:
                    break
        
        # Perform Lookahead
        for cand in lookahead_candidates:
            rem_set = cand['new_rem']
            if not rem_set:
                cand['final_score'] = cand['base_score']
                continue
            
            # Find Top-N LPT targets
            targets = []
            c = 0
            for t in sorted_lpt:
                if t in rem_set:
                    targets.append(t)
                    c += 1
                    if c >= LOOKAHEAD_TARGETS: break
            
            if targets:
                # Optimistic Lookahead:
                # What is the BEST score we can achieve by adding one of the top heavy items next?
                best_la = float('inf')
                for next_t in targets:
                    la_seq = cand['seq'] + [next_t]
                    la_cost = workload.get_opt_seq_cost(la_seq)
                    la_total = cand['total_dur'] + txn_durations[next_t]
                    score = la_cost - la_total
                    if score < best_la:
                        best_la = score
                cand['final_score'] = best_la
            else:
                cand['final_score'] = cand['base_score']
                
        # Final Selection for Next Beam
        lookahead_candidates.sort(key=lambda x: (x['final_score'], -x['total_dur']))
        
        new_beam = []
        parent_counts = {i: 0 for i in range(len(beam))}
        
        for cand in lookahead_candidates:
            if len(new_beam) >= BEAM_WIDTH: break
            
            p_idx = cand['p_idx']
            if parent_counts[p_idx] < MAX_CHILDREN_PER_PARENT:
                new_beam.append({
                    'cost': cand['cost'],
                    'total_dur': cand['total_dur'],
                    'seq': cand['seq'],
                    'rem': cand['new_rem']
                })
                parent_counts[p_idx] += 1
        
        if not new_beam: break
        beam = new_beam

    # Extract Best
    best_node = min(beam, key=lambda x: x['cost'])
    best_cost = best_node['cost']
    best_seq = best_node['seq']

    # --- Iterated Local Search (ILS) ---
    
    def local_search(seq, current_cost):
        """
        Applies Insertions, Block Moves, and Swaps to improve the sequence.
        Returns (seq, cost).
        """
        improved = True
        while improved:
            improved = False
            
            # 1. Insertions (Window 10 - Broad search for placement)
            w_ins = 10
            for i in range(len(seq)):
                start = max(0, i - w_ins)
                end = min(len(seq), i + w_ins)
                if start >= end: continue
                
                curr = seq[i]
                temp = seq[:i] + seq[i+1:]
                
                # Check positions in window
                for k in range(start, end):
                    if k == i: continue
                    cand = temp[:k] + [curr] + temp[k:]
                    c = workload.get_opt_seq_cost(cand)
                    if c < current_cost:
                        current_cost = c
                        seq = cand
                        improved = True
                        break
                if improved: break
            if improved: continue

            # 2. Block Moves (Size 2, Window 10)
            # Moves adjacent pairs. Good for dependent pairs.
            w_blk = 10
            if len(seq) > 4:
                for i in range(len(seq) - 1):
                    block = seq[i:i+2]
                    temp = seq[:i] + seq[i+2:]
                    
                    start = max(0, i - w_blk)
                    end = min(len(temp) + 1, i + w_blk)
                    
                    for k in range(start, end):
                        # Skip original position (approx)
                        if abs(k - i) < 2: continue
                        
                        cand = temp[:k] + block + temp[k:]
                        c = workload.get_opt_seq_cost(cand)
                        if c < current_cost:
                            current_cost = c
                            seq = cand
                            improved = True
                            break
                    if improved: break
            if improved: continue

            # 3. Swaps (Window 5) - Fine-tuning
            for i in range(len(seq) - 1):
                for offset in range(1, 6):
                    j = i + offset
                    if j >= len(seq): break
                    
                    # Swap
                    seq[i], seq[j] = seq[j], seq[i]
                    c = workload.get_opt_seq_cost(seq)
                    if c < current_cost:
                        current_cost = c
                        improved = True
                    else:
                        # Revert
                        seq[i], seq[j] = seq[j], seq[i]
                if improved: break
                
        return seq, current_cost

    # Initial refinement of Beam Search result
    best_seq, best_cost = local_search(best_seq, best_cost)

    # ILS Loop
    curr_seq = best_seq[:]
    curr_cost = best_cost
    
    for cycle in range(ILS_CYCLES):
        # Calculate Acceptance Threshold
        # Linear decay from START to 0
        threshold = THRESHOLD_START * (1.0 - (cycle / ILS_CYCLES))
        if threshold < 0: threshold = 0
        
        p_seq = curr_seq[:]
        
        # --- Ruin ---
        # Remove a significant chunk to escape local optima
        if len(p_seq) > RUIN_MAX + 2:
            ruin_size = random.randint(RUIN_MIN, RUIN_MAX)
            removed = []
            for _ in range(ruin_size):
                if not p_seq: break
                idx = random.randint(0, len(p_seq) - 1)
                removed.append(p_seq.pop(idx))
            
            # --- Recreate ---
            # Greedy Best-Fit: Insert heaviest items first into best position
            removed.sort(key=lambda t: txn_durations[t], reverse=True)
            
            for t in removed:
                best_pos = -1
                best_c = float('inf')
                # Try all positions
                for pos in range(len(p_seq) + 1):
                    cand = p_seq[:pos] + [t] + p_seq[pos:]
                    c = workload.get_opt_seq_cost(cand)
                    if c < best_c:
                        best_c = c
                        best_pos = pos
                p_seq.insert(best_pos, t)
                
        # --- Repair ---
        p_seq, p_cost = local_search(p_seq, workload.get_opt_seq_cost(p_seq))
        
        # --- Acceptance ---
        # 1. New Global Best
        if p_cost < best_cost:
            best_cost = p_cost
            best_seq = p_seq
            curr_seq = p_seq
            curr_cost = p_cost
        # 2. Threshold Acceptance (Exploration)
        elif p_cost < curr_cost * (1.0 + threshold):
            curr_seq = p_seq
            curr_cost = p_cost
        # 3. Rejection (Implicitly keep curr_seq)

    return best_cost, best_seq


def get_random_costs():
    """
    Evaluate scheduling algorithm on three different workloads.

    Returns:
        Tuple of (total_makespan, list_of_schedules, execution_time)
    """
    start_time = time.time()

    # Effort Level (scales beam width)
    # Increased slightly to allow broader search
    effort_level = 12

    workload1 = Workload(WORKLOAD_1)
    makespan1, schedule1 = get_best_schedule(workload1, effort_level)

    workload2 = Workload(WORKLOAD_2)
    makespan2, schedule2 = get_best_schedule(workload2, effort_level)

    workload3 = Workload(WORKLOAD_3)
    makespan3, schedule3 = get_best_schedule(workload3, effort_level)

    total_makespan = makespan1 + makespan2 + makespan3
    schedules = [schedule1, schedule2, schedule3]
    execution_time = time.time() - start_time

    return total_makespan, schedules, execution_time
