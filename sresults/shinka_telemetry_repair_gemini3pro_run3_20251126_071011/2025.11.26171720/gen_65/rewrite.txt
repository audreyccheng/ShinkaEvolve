from typing import Dict, Any, Tuple, List
import math

def repair_network_telemetry(telemetry: Dict[str, Dict[str, Any]], 
                             topology: Dict[str, List[str]]) -> Dict[str, Dict[str, Tuple]]:
    """
    Repairs network telemetry using a Status-Aware Bayesian Flow Consensus algorithm.
    
    Key Features:
    1.  **Anchor Strength**: Routers are assigned a reliability score based on the ratio of 
        consistent (Anchor) traffic to total traffic.
    2.  **Status Priors**: Explicitly incorporates interface status into hypothesis generation,
        strongly favoring zero flow for 'down' interfaces.
    3.  **Adaptive External Validation**: External links are repaired based on router anchor strength;
        high strength routers effectively override external measurements with conservation-derived values.
    4.  **Mid-Course Locking**: Promotes high-confidence suspect flows to anchors halfway through 
        iterations to stabilize convergence.
    5.  **Margin-based Confidence**: Confidence is derived from the probability mass of the winning 
        hypothesis cluster, scaled by the router's conservation fit.
    """
    
    # --- Configuration ---
    SYMMETRY_TOLERANCE = 0.02
    CONSERVATION_TOLERANCE_PCT = 0.02
    MIN_SIGNIFICANT_FLOW = 0.5
    ITERATIONS = 5
    MOMENTUM = 0.6
    LOCKING_THRESHOLD = 0.95  # Probability threshold to lock a suspect flow
    LOCKING_ITERATION = 3     # Iteration to perform locking
    
    # --- Helper Structures ---
    if_to_router = {}
    for r_id, if_list in topology.items():
        for i_id in if_list:
            if_to_router[i_id] = r_id
            
    # Estimates state: {if_id: {'rx': val, 'tx': val}}
    estimates = {}
    
    # Links classification
    links = {}
    processed_ifs = set()
    
    for if_id, data in telemetry.items():
        estimates[if_id] = {
            'rx': data.get('rx_rate', 0.0), 
            'tx': data.get('tx_rate', 0.0)
        }
        
        if if_id in processed_ifs: continue
        
        peer = data.get('connected_to')
        if peer and peer in telemetry:
            link_key = tuple(sorted([if_id, peer]))
            links[link_key] = {'type': 'internal', 'if1': if_id, 'if2': peer}
            processed_ifs.add(if_id)
            processed_ifs.add(peer)
        else:
            links[(if_id,)] = {'type': 'external', 'if1': if_id, 'if2': None}
            processed_ifs.add(if_id)
            
    # --- Step 1: Initial Symmetry & Classification ---
    
    suspect_flows = [] 
    anchors = set()    # Set of (if_id, metric) that are trusted
    
    # We also track which flows are "hard" locked (initially anchors, later promoted suspects)
    locked_flows = set() 
    
    for key, info in links.items():
        if1 = info['if1']
        
        if info['type'] == 'internal':
            if2 = info['if2']
            
            # Helper to check symmetry
            def check_sym(v1, v2):
                diff = abs(v1 - v2)
                denom = max(v1, v2, 1.0)
                return (diff / denom < SYMMETRY_TOLERANCE) or (diff < 1.0)

            # Forward: 1(TX) -> 2(RX)
            t1, r2 = estimates[if1]['tx'], estimates[if2]['rx']
            if check_sym(t1, r2):
                avg = (t1 + r2) / 2.0
                estimates[if1]['tx'] = avg
                estimates[if2]['rx'] = avg
                anchors.add((if1, 'tx'))
                anchors.add((if2, 'rx'))
                locked_flows.add((if1, 'tx'))
                locked_flows.add((if2, 'rx'))
            else:
                suspect_flows.append({
                    'type': 'internal', 'src': if1, 'dst': if2,
                    'cands': [t1, r2],
                    'status_src': telemetry[if1].get('interface_status', 'unknown'),
                    'status_dst': telemetry[if2].get('interface_status', 'unknown')
                })
                
            # Backward: 2(TX) -> 1(RX)
            t2, r1 = estimates[if2]['tx'], estimates[if1]['rx']
            if check_sym(t2, r1):
                avg = (t2 + r1) / 2.0
                estimates[if2]['tx'] = avg
                estimates[if1]['rx'] = avg
                anchors.add((if2, 'tx'))
                anchors.add((if1, 'rx'))
                locked_flows.add((if2, 'tx'))
                locked_flows.add((if1, 'rx'))
            else:
                suspect_flows.append({
                    'type': 'internal', 'src': if2, 'dst': if1,
                    'cands': [t2, r1],
                    'status_src': telemetry[if2].get('interface_status', 'unknown'),
                    'status_dst': telemetry[if1].get('interface_status', 'unknown')
                })
        else:
            # External
            suspect_flows.append({
                'type': 'external', 'src': if1, 'dst': None,
                'cands': [estimates[if1]['tx']],
                'metric': 'tx',
                'status_src': telemetry[if1].get('interface_status', 'unknown')
            })
            suspect_flows.append({
                'type': 'external', 'src': None, 'dst': if1,
                'cands': [estimates[if1]['rx']],
                'metric': 'rx',
                'status_dst': telemetry[if1].get('interface_status', 'unknown')
            })

    # --- Step 2: Iterative Solver ---
    
    def get_router_state(rid):
        if rid not in topology: return 0.0, 1.0
        tin, tout = 0.0, 0.0
        max_f = 0.0
        for iid in topology[rid]:
            if iid in estimates:
                r, t = estimates[iid]['rx'], estimates[iid]['tx']
                tin += r
                tout += t
                max_f = max(max_f, r, t)
        return (tin - tout), max(max_f, 1.0)
        
    def calc_sigma(flow_val):
        # Adaptive noise model
        return max(math.sqrt(flow_val), flow_val * CONSERVATION_TOLERANCE_PCT, 1.0)

    # Store probability of winning hypothesis for confidence calibration
    flow_confidence = {} 
    
    for iter_idx in range(ITERATIONS):
        
        # 2a. Calculate Router Anchor Strength
        # This determines how much we trust "Implied" values from conservation
        router_anchor_strength = {}
        for rid in topology:
            anchor_flow = 0.0
            total_flow = 0.0
            for iid in topology[rid]:
                if iid in estimates:
                    r, t = estimates[iid]['rx'], estimates[iid]['tx']
                    total_flow += (r + t)
                    if (iid, 'rx') in locked_flows: anchor_flow += r
                    if (iid, 'tx') in locked_flows: anchor_flow += t
            
            ratio = anchor_flow / max(total_flow, 1.0)
            # Sigmoid-like scaling to push unclear cases to middle
            router_anchor_strength[rid] = ratio

        updates = []
        flows_to_lock = []

        for f_idx, flow in enumerate(suspect_flows):
            # Skip if already processed/locked (though we iterate suspect_flows, we can mark them)
            if flow.get('locked', False): continue

            if flow['type'] == 'internal':
                src, dst = flow['src'], flow['dst']
                r_src = if_to_router.get(src)
                r_dst = if_to_router.get(dst)
                
                s_src = flow.get('status_src')
                s_dst = flow.get('status_dst')
                
                # Hypotheses
                cands = [c for c in flow['cands'] if c >= 0]
                if len(cands) == 2:
                    v1, v2 = cands
                    if abs(v1-v2) < max(v1,v2)*0.1 + 1.0: cands.append((v1+v2)/2.0)
                
                hyps = sorted(list(set(cands + [0.0])))
                
                # Save current
                curr_tx = estimates[src]['tx']
                curr_rx = estimates[dst]['rx']
                
                scores = []
                for h in hyps:
                    estimates[src]['tx'] = h
                    estimates[dst]['rx'] = h
                    
                    # Likelihoods
                    imb_s, flow_s = get_router_state(r_src)
                    score_s = math.exp(-abs(imb_s)/calc_sigma(flow_s)) if r_src else 1.0
                    
                    imb_d, flow_d = get_router_state(r_dst)
                    score_d = math.exp(-abs(imb_d)/calc_sigma(flow_d)) if r_dst else 1.0
                    
                    # Prior
                    prior = 1.0
                    if h == 0.0:
                        # Status-aware prior
                        if s_src == 'down' or s_dst == 'down': prior = 0.99
                        elif max(flow['cands']) > 5.0: prior = 0.01 # Unlikely to be 0 if measured high and UP
                        else: prior = 0.5
                    else:
                        # Gaussian prior around measurements
                        # If h is close to ANY candidate, it gets a boost
                        dist = min([abs(h - c) for c in cands])
                        prior = math.exp(-dist / max(h*0.05, 1.0))
                    
                    scores.append(score_s * score_d * prior)
                
                estimates[src]['tx'] = curr_tx
                estimates[dst]['rx'] = curr_rx
                
                # Selection
                total = sum(scores) + 1e-20
                probs = [s/total for s in scores]
                best_idx = scores.index(max(scores))
                win_val = hyps[best_idx]
                win_prob = probs[best_idx]
                
                # Aggregate probability for cluster around winner
                cluster_prob = sum(p for i, p in enumerate(probs) 
                                 if abs(hyps[i] - win_val) < max(win_val*0.05, 1.0))
                
                updates.append((src, 'tx', win_val, cluster_prob))
                updates.append((dst, 'rx', win_val, cluster_prob))
                
                # Check for locking
                if iter_idx == LOCKING_ITERATION and cluster_prob > LOCKING_THRESHOLD:
                    flows_to_lock.append(f_idx)
                    locked_flows.add((src, 'tx'))
                    locked_flows.add((dst, 'rx'))
                    
            elif flow['type'] == 'external':
                if flow['src']:
                    if_id = flow['src']
                    r_id = if_to_router.get(if_id)
                    metric = 'tx'
                    meas = flow['cands'][0]
                    curr_val = estimates[if_id]['tx']
                    stat = flow.get('status_src')
                    
                    imb, r_flow = get_router_state(r_id)
                    implied = max(0.0, curr_val + imb)
                else:
                    if_id = flow['dst']
                    r_id = if_to_router.get(if_id)
                    metric = 'rx'
                    meas = flow['cands'][0]
                    curr_val = estimates[if_id]['rx']
                    stat = flow.get('status_dst')
                    
                    imb, r_flow = get_router_state(r_id)
                    implied = max(0.0, curr_val - imb)
                
                hyps = sorted(list(set([meas, implied, 0.0])))
                scores = []
                
                anchor_strength = router_anchor_strength.get(r_id, 0.5)
                
                for h in hyps:
                    estimates[if_id][metric] = h
                    imb, rf = get_router_state(r_id)
                    lik = math.exp(-abs(imb)/calc_sigma(rf)) if r_id else 0.5
                    
                    prior = 1.0
                    # Prior logic for External
                    if h == 0.0:
                        if stat == 'down': prior = 0.99
                        elif meas > 10.0: prior = 0.01
                        else: prior = 0.5
                    elif abs(h - implied) < 0.001:
                        # This is the "Conservation is Correct" hypothesis
                        # Trust it proportional to anchor strength
                        # Base trust 0.3, max 0.95
                        prior = 0.3 + (0.65 * anchor_strength)
                    elif abs(h - meas) < 0.001:
                        # This is the "Measurement is Correct" hypothesis
                        # Trust drops if anchor strength is high (implies conservation is reliable)
                        prior = 0.8 - (0.5 * anchor_strength)
                    
                    scores.append(lik * prior)
                
                estimates[if_id][metric] = curr_val
                
                total = sum(scores) + 1e-20
                probs = [s/total for s in scores]
                best_idx = scores.index(max(scores))
                win_val = hyps[best_idx]
                
                cluster_prob = sum(p for i, p in enumerate(probs) 
                                 if abs(hyps[i] - win_val) < max(win_val*0.05, 1.0))
                
                updates.append((if_id, metric, win_val, cluster_prob))

        # Apply Updates
        for if_id, metric, val, prob in updates:
            if (if_id, metric) in locked_flows and iter_idx > LOCKING_ITERATION:
                continue # Don't update locked flows
            
            old = estimates[if_id][metric]
            estimates[if_id][metric] = (old * (1 - MOMENTUM)) + (val * MOMENTUM)
            flow_confidence[(if_id, metric)] = prob
            
        # Mark locked flows
        for idx in flows_to_lock:
            suspect_flows[idx]['locked'] = True

    # --- Step 3: Confidence & Status Output ---
    
    result = {}
    
    # Calculate final router fits
    router_fits = {}
    for rid in topology:
        imb, flow = get_router_state(rid)
        fit = math.exp(-abs(imb) / calc_sigma(flow))
        router_fits[rid] = fit

    for if_id, data in telemetry.items():
        orig_rx = data.get('rx_rate', 0.0)
        orig_tx = data.get('tx_rate', 0.0)
        orig_status = data.get('interface_status', 'unknown')
        
        rep_rx = estimates[if_id]['rx']
        rep_tx = estimates[if_id]['tx']
        
        rid = if_to_router.get(if_id)
        r_fit = router_fits.get(rid, 0.8)
        
        def get_conf(metric, val, orig):
            # 1. Base confidence from solver (margin of victory)
            sol_prob = flow_confidence.get((if_id, metric), 0.5)
            
            # 2. Adjust based on Anchor status
            if (if_id, metric) in anchors:
                base_conf = 0.98
            else:
                # If we changed the value significantly, we rely on the solver's confidence
                # scaled by how well the router actually balanced.
                base_conf = sol_prob * (0.3 + 0.7 * r_fit)
            
            # 3. Penalty for contradicting status without overwhelming evidence
            if val > MIN_SIGNIFICANT_FLOW and orig_status == 'down':
                 # We are saying it's UP when it says DOWN.
                 # Only trust this if solver was very sure and fit is great.
                 base_conf *= 0.9
            
            return max(0.01, min(0.99, base_conf))

        conf_rx = get_conf('rx', rep_rx, orig_rx)
        conf_tx = get_conf('tx', rep_tx, orig_tx)
        
        # Status Inference
        peer_id = data.get('connected_to')
        peer_status = 'unknown'
        if peer_id and peer_id in telemetry:
            peer_status = telemetry[peer_id].get('interface_status', 'unknown')
            
        has_traffic = (rep_rx > MIN_SIGNIFICANT_FLOW) or (rep_tx > MIN_SIGNIFICANT_FLOW)
        
        rep_status = orig_status
        conf_status = 1.0
        
        if has_traffic:
            rep_status = 'up'
            if orig_status != 'up':
                conf_status = (conf_rx + conf_tx) / 2.0
        elif peer_status == 'down':
            rep_status = 'down'
            if orig_status != 'down':
                conf_status = 0.95
        else:
            # Ambiguous (No traffic, Peer UP/Unknown, Local UP/Unknown)
            # If orig is UP, likely idle -> UP.
            # If orig is DOWN, confirmed -> DOWN.
            rep_status = orig_status
            
        # Consistency enforcement
        if rep_status == 'down':
            rep_rx, rep_tx = 0.0, 0.0
            conf_rx = max(conf_rx, 0.95)
            conf_tx = max(conf_tx, 0.95)
            
        entry = {}
        entry['rx_rate'] = (orig_rx, rep_rx, conf_rx)
        entry['tx_rate'] = (orig_tx, rep_tx, conf_tx)
        entry['interface_status'] = (orig_status, rep_status, conf_status)
        for k in ['connected_to', 'local_router', 'remote_router']:
            if k in data: entry[k] = data[k]
        result[if_id] = entry
        
    return result
