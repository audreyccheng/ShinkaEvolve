"""
Expert parallelism load balancer (EPLB) for vLLM.

This module implements the core rearrangement algorithm.

The rearrangement algorithm is adapted from
[DeepSeek EPLB](https://github.com/deepseek-ai/eplb).

Please find at [#12](https://github.com/deepseek-ai/EPLB/issues/12) an example
on how the EPLB algorithm works.
"""

import torch


def balanced_packing(weight: torch.Tensor,
                     num_packs: int) -> tuple[torch.Tensor, torch.Tensor]:
    """
    Pack n weighted objects to m packs, such that each bin contains exactly
    n/m objects and the weights of all packs are as balanced as possible.

    Uses Hybrid ZigZag/Greedy Initialization with Vectorized Refinement.

    Parameters:
        weight: [X, n], the weight of each item
        num_packs: number of packs

    Returns:
        pack_index: [X, n], the pack index of each item
        rank_in_pack: [X, n], the rank of the item in the pack
    """
    num_layers, num_groups = weight.shape
    device = weight.device
    assert num_groups % num_packs == 0
    groups_per_pack = num_groups // num_packs

    if groups_per_pack == 1:
        pack_index = torch.arange(weight.size(-1),
                                  dtype=torch.int64,
                                  device=device).expand(weight.shape)
        rank_in_pack = torch.zeros_like(weight, dtype=torch.int64)
        return pack_index, rank_in_pack

    # --- Hyperparameters ---
    # Generate many cheap solutions (ZigZag is nearly free), prune, then refine.
    NUM_RESTARTS = 128
    NUM_REFINED = 16
    
    # --- 1. Initialization (Hybrid: ZigZag + Randomized Greedy) ---
    
    # We allocate 50% of restarts to ZigZag (Snake) and 50% to Greedy LPT
    n_zigzag = NUM_RESTARTS // 2
    n_greedy = NUM_RESTARTS - n_zigzag
    
    # Replicate weights: [L, N, G]
    w_base = weight.unsqueeze(1).repeat(1, NUM_RESTARTS, 1)
    
    # Add multiplicative noise to diverse generation
    # Keep the first instance of each strategy 'clean' (no noise)
    noise = torch.rand_like(w_base) * 0.15
    w_noisy = w_base * (1.0 + noise)
    w_noisy[:, 0, :] = w_base[:, 0, :]
    w_noisy[:, n_zigzag, :] = w_base[:, n_zigzag, :]
    
    # Flatten for batch processing: [B, G] where B = L * N
    flat_w = w_noisy.view(-1, num_groups)
    flat_orig_w = w_base.view(-1, num_groups)
    
    # Sort weights descending
    sorted_w, sorted_idx = flat_w.sort(dim=1, descending=True)
    # Gather original weights in the sorted order for accurate accounting
    sorted_orig_w = torch.gather(flat_orig_w, 1, sorted_idx)
    
    batch_size = flat_w.shape[0]
    
    # Container for initial assignments [B, G] -> value is pack_id
    pack_assignment = torch.empty(batch_size, num_groups, dtype=torch.int64, device=device)
    
    # --- Strategy A: ZigZag (Snake) ---
    # Assign packs in a pattern: 0, 1, ... M-1, M-1, ... 1, 0, 0 ...
    # This works well for sorted data.
    zigzag_mask = torch.zeros(NUM_RESTARTS, dtype=torch.bool, device=device)
    zigzag_mask[:n_zigzag] = True
    zigzag_mask = zigzag_mask.repeat(num_layers) # Expand to batch
    
    if n_zigzag > 0:
        ranks = torch.arange(num_groups, device=device)
        period = 2 * num_packs
        pattern = ranks % period
        # Fold the triangle wave: if x >= M, x = 2M - 1 - x
        pattern = torch.where(pattern < num_packs, pattern, period - 1 - pattern)
        pack_assignment[zigzag_mask] = pattern.unsqueeze(0)
        
    # --- Strategy B: Greedy LPT ---
    # Place item in the currently lightest valid pack
    greedy_mask = ~zigzag_mask
    if n_greedy > 0:
        g_w = sorted_orig_w[greedy_mask]
        num_greedy_batch = g_w.shape[0]
        
        p_weights = torch.zeros(num_greedy_batch, num_packs, device=device)
        p_counts = torch.zeros(num_greedy_batch, num_packs, dtype=torch.int64, device=device)
        g_assign = torch.zeros(num_greedy_batch, num_groups, dtype=torch.int64, device=device)
        
        inf_tensor = torch.tensor(float('inf'), device=device)
        row_indices = torch.arange(num_greedy_batch, device=device)
        
        for i in range(num_groups):
            val = g_w[:, i]
            # Mask packs that are full
            valid_mask = p_counts < groups_per_pack
            # Find min weight among valid
            w_masked = torch.where(valid_mask, p_weights, inf_tensor)
            chosen_pack = w_masked.argmin(dim=1)
            
            # Update
            p_weights[row_indices, chosen_pack] += val
            p_counts[row_indices, chosen_pack] += 1
            g_assign[:, i] = chosen_pack
            
        pack_assignment[greedy_mask] = g_assign

    # --- 2. Pruning ---
    # Calculate imbalance for all candidates
    p_weights_all = torch.zeros(batch_size, num_packs, device=device)
    p_weights_all.scatter_add_(1, pack_assignment, sorted_orig_w)
    
    imbalance = p_weights_all.max(dim=1).values - p_weights_all.min(dim=1).values
    imbalance = imbalance.view(num_layers, NUM_RESTARTS)
    
    # Select top K candidates per layer
    _, best_indices = imbalance.topk(NUM_REFINED, dim=1, largest=False)
    
    # Flatten selection
    offsets = (torch.arange(num_layers, device=device) * NUM_RESTARTS).unsqueeze(1)
    flat_selected = (offsets + best_indices).flatten()
    
    refined_batch_size = flat_selected.size(0)
    
    # Gather selected data
    sel_assignment = pack_assignment[flat_selected] # [B', G]
    sel_sorted_w = sorted_orig_w[flat_selected]     # [B', G]
    sel_sorted_idx = sorted_idx[flat_selected]      # [B', G]
    
    # Convert to structured Pack Contents [B', Packs, Slots]
    # We sort the assignment to group items by pack.
    # Since items are already sorted by weight (descending), stable sort preserves this order within packs.
    pack_sort_val, pack_sort_idx = sel_assignment.sort(dim=1, stable=True)
    
    ordered_w = torch.gather(sel_sorted_w, 1, pack_sort_idx)
    ordered_ids = torch.gather(sel_sorted_idx, 1, pack_sort_idx)
    
    pack_contents = ordered_w.view(refined_batch_size, num_packs, groups_per_pack)
    pack_item_ids = ordered_ids.view(refined_batch_size, num_packs, groups_per_pack)
    
    # --- 3. Refinement (1-for-1 and 2-for-2 Swaps) ---
    
    # Pre-compute pairs for 2-for-2 if feasible
    check_2for2 = (groups_per_pack >= 2) and (groups_per_pack <= 32)
    if check_2for2:
        r_idx, c_idx = torch.triu_indices(groups_per_pack, groups_per_pack, offset=1, device=device)
    
    b_idx = torch.arange(refined_batch_size, device=device)
    
    for _ in range(50):
        # Current Pack Weights
        curr_w = pack_contents.sum(dim=2)
        
        val_max, p_max = curr_w.max(dim=1)
        val_min, p_min = curr_w.min(dim=1)
        
        diff = val_max - val_min
        active = diff > 1e-4
        if not active.any():
            break
            
        # Extract items from max and min packs
        # [B, S]
        items_max = pack_contents[b_idx, p_max]
        items_min = pack_contents[b_idx, p_min]
        
        # --- 1-for-1 Analysis ---
        # delta = max - min
        # improvement = diff - |diff - 2*delta|
        # [B, S, S]
        delta_1 = items_max.unsqueeze(2) - items_min.unsqueeze(1)
        
        target = diff.view(-1, 1, 1)
        imp_1 = target - (target - 2 * delta_1).abs()
        
        # Filter: must strictly reduce max pack weight
        mask_1 = (delta_1 > 1e-6)
        imp_1[~mask_1] = -1.0
        
        best_imp_1, best_idx_1 = imp_1.view(refined_batch_size, -1).max(dim=1)
        
        # --- 2-for-2 Analysis ---
        best_imp_2 = torch.full_like(best_imp_1, -1.0)
        best_idx_2 = torch.zeros_like(best_idx_1)
        
        if check_2for2:
            # Pair sums: [B, N_pairs]
            pair_max = items_max[:, r_idx] + items_max[:, c_idx]
            pair_min = items_min[:, r_idx] + items_min[:, c_idx]
            
            # [B, N_pairs, N_pairs]
            delta_2 = pair_max.unsqueeze(2) - pair_min.unsqueeze(1)
            imp_2 = target - (target - 2 * delta_2).abs()
            
            mask_2 = (delta_2 > 1e-6)
            imp_2[~mask_2] = -1.0
            
            best_imp_2, best_idx_2 = imp_2.view(refined_batch_size, -1).max(dim=1)
        
        # --- Apply Best Move ---
        use_2 = (best_imp_2 > best_imp_1) & (best_imp_2 > 1e-6)
        use_1 = (~use_2) & (best_imp_1 > 1e-6)
        
        do_any = (use_1 | use_2) & active
        if not do_any.any():
            break
            
        # Apply 1-for-1
        if use_1.any():
            bb = b_idx[use_1]
            idx = best_idx_1[use_1]
            
            u = idx // groups_per_pack
            v = idx % groups_per_pack
            
            pm = p_max[use_1]
            pn = p_min[use_1]
            
            val_u = pack_contents[bb, pm, u]
            val_v = pack_contents[bb, pn, v]
            pack_contents[bb, pm, u] = val_v
            pack_contents[bb, pn, v] = val_u
            
            id_u = pack_item_ids[bb, pm, u]
            id_v = pack_item_ids[bb, pn, v]
            pack_item_ids[bb, pm, u] = id_v
            pack_item_ids[bb, pn, v] = id_u
            
        # Apply 2-for-2
        if use_2.any():
            bb = b_idx[use_2]
            idx = best_idx_2[use_2]
            n_pairs = r_idx.shape[0]
            
            pair_u = idx // n_pairs
            pair_v = idx % n_pairs
            
            u1, u2 = r_idx[pair_u], c_idx[pair_u]
            v1, v2 = r_idx[pair_v], c_idx[pair_v]
            
            pm = p_max[use_2]
            pn = p_min[use_2]
            
            # Swap pair u -> v
            # Item 1
            val_u1 = pack_contents[bb, pm, u1]
            val_v1 = pack_contents[bb, pn, v1]
            pack_contents[bb, pm, u1] = val_v1
            pack_contents[bb, pn, v1] = val_u1
            
            id_u1 = pack_item_ids[bb, pm, u1]
            id_v1 = pack_item_ids[bb, pn, v1]
            pack_item_ids[bb, pm, u1] = id_v1
            pack_item_ids[bb, pn, v1] = id_u1
            
            # Item 2
            val_u2 = pack_contents[bb, pm, u2]
            val_v2 = pack_contents[bb, pn, v2]
            pack_contents[bb, pm, u2] = val_v2
            pack_contents[bb, pn, v2] = val_u2
            
            id_u2 = pack_item_ids[bb, pm, u2]
            id_v2 = pack_item_ids[bb, pn, v2]
            pack_item_ids[bb, pm, u2] = id_v2
            pack_item_ids[bb, pn, v2] = id_u2

    # --- 4. Final Selection ---
    final_w = pack_contents.sum(dim=2)
    final_diff = final_w.max(dim=1).values - final_w.min(dim=1).values
    final_diff = final_diff.view(num_layers, NUM_REFINED)
    
    best_restart = final_diff.argmin(dim=1)
    
    # Gather Final Indices
    # [L]
    chosen_flat_idx = torch.arange(num_layers, device=device) * NUM_REFINED + best_restart
    
    # [L, P, S]
    final_ids = pack_item_ids.view(num_layers, NUM_REFINED, num_packs, groups_per_pack)[
        torch.arange(num_layers, device=device), best_restart
    ]
    
    # Reconstruct Output Tensors
    pack_index = torch.empty(num_layers, num_groups, dtype=torch.int64, device=device)
    rank_in_pack = torch.empty(num_layers, num_groups, dtype=torch.int64, device=device)
    
    # Generate grids for scatter
    grid_pack = torch.arange(num_packs, device=device).view(1, -1, 1).expand(num_layers, -1, groups_per_pack)
    grid_rank = torch.arange(groups_per_pack, device=device).view(1, 1, -1).expand(num_layers, num_packs, -1)
    
    flat_item_ids = final_ids.view(num_layers, -1)
    flat_pack_vals = grid_pack.reshape(num_layers, -1)
    flat_rank_vals = grid_rank.reshape(num_layers, -1)
    
    pack_index.scatter_(1, flat_item_ids, flat_pack_vals)
    rank_in_pack.scatter_(1, flat_item_ids, flat_rank_vals)
    
    return pack_index, rank_in_pack


def replicate_experts(
        weight: torch.Tensor,
        num_phy: int) -> tuple[torch.Tensor, torch.Tensor, torch.Tensor]:
    """
    Replicate `num_log` experts to `num_phy` replicas, such that the maximum
    load of all replicas is minimized.

    Parameters:
        weight: [X, num_log]
        num_phy: total number of experts after replication

    Returns:
        phy2log: [X, num_phy], logical expert id of each physical expert
        rank: [X, num_phy], the replica rank
        logcnt: [X, num_log], number of replicas for each logical expert
    """
    n, num_log = weight.shape
    num_redundant = num_phy - num_log
    assert num_redundant >= 0
    device = weight.device
    phy2log = torch.arange(num_phy, dtype=torch.int64,
                           device=device).repeat(n, 1)
    rank = torch.zeros(n, num_phy, dtype=torch.int64, device=device)
    logcnt = torch.ones(n, num_log, dtype=torch.int64, device=device)
    arangen = torch.arange(n, dtype=torch.int64, device=device)

    # Pre-compute scores to avoid redundant division
    current_scores = weight.float() / logcnt.float()

    for i in range(num_log, num_phy):
        redundant_indices = current_scores.argmax(dim=-1)
        phy2log[:, i] = redundant_indices
        rank[:, i] = logcnt[arangen, redundant_indices]

        # Update logcnt
        logcnt[arangen, redundant_indices] += 1

        # Incrementally update scores only for modified experts
        new_cnt = logcnt[arangen, redundant_indices].float()
        chosen_weight = weight[arangen, redundant_indices].float()
        current_scores[arangen, redundant_indices] = chosen_weight / new_cnt

    return phy2log, rank, logcnt


def rebalance_experts_hierarchical(
    weight: torch.Tensor,
    num_physical_experts: int,
    num_groups: int,
    num_nodes: int,
    num_gpus: int,
):
    """
    Parameters:
        weight: [num_moe_layers, num_logical_experts]
        num_physical_experts: number of physical experts after replication
        num_groups: number of expert groups
        num_nodes: number of server nodes, where the intra-node network
        (e.g, NVLink) is faster
        num_gpus: number of GPUs, must be a multiple of `num_nodes`

    Returns:
        physical_to_logical_map: [num_moe_layers, num_physical_experts]
        logical_to_physical_map: [num_moe_layers, num_logical_experts, X]
        logical_count: [num_moe_layers, num_logical_experts]
    """
    num_layers, num_logical_experts = weight.shape
    assert num_logical_experts % num_groups == 0
    group_size = num_logical_experts // num_groups
    assert num_groups % num_nodes == 0
    groups_per_node = num_groups // num_nodes
    assert num_gpus % num_nodes == 0
    assert num_physical_experts % num_gpus == 0
    phy_experts_per_gpu = num_physical_experts // num_gpus

    def inverse(perm: torch.Tensor) -> torch.Tensor:
        inv = torch.empty_like(perm)
        inv.scatter_(
            1,
            perm,
            torch.arange(perm.size(1), dtype=torch.int64,
                         device=perm.device).expand(perm.shape),
        )
        return inv

    # Step 1: pack groups to nodes
    tokens_per_group = weight.unflatten(-1, (num_groups, group_size)).sum(-1)
    group_pack_index, group_rank_in_pack = balanced_packing(
        tokens_per_group, num_nodes)
    log2mlog = (((group_pack_index * groups_per_node + group_rank_in_pack) *
                 group_size).unsqueeze(-1) +
                torch.arange(group_size,
                             dtype=torch.int64,
                             device=group_pack_index.device)).flatten(-2)
    mlog2log = inverse(log2mlog)

    # Step 2: construct redundant experts within nodes
    # [num_layers * num_nodes, num_logical_experts // num_nodes]
    tokens_per_mlog = weight.gather(-1, mlog2log).view(
        -1, num_logical_experts // num_nodes)
    phy2mlog, phyrank, mlogcnt = replicate_experts(
        tokens_per_mlog, num_physical_experts // num_nodes)

    # Step 3: pack physical_experts to GPUs
    # [num_layers * num_nodes, num_physical_experts // num_nodes]
    tokens_per_phy = (tokens_per_mlog / mlogcnt).gather(-1, phy2mlog)
    pack_index, rank_in_pack = balanced_packing(tokens_per_phy,
                                                num_gpus // num_nodes)
    phy2pphy = pack_index * phy_experts_per_gpu + rank_in_pack
    pphy2phy = inverse(phy2pphy)

    pphy2mlog = phy2mlog.gather(
        -1, pphy2phy)  # [num_layers * num_nodes, num_log_per_nodes]
    pphy2mlog = (pphy2mlog.view(num_layers, num_nodes, -1) + torch.arange(
        0,
        num_logical_experts,
        num_logical_experts // num_nodes,
        device=group_pack_index.device,
    ).view(1, -1, 1)).flatten(-2)
    pphy2log = mlog2log.gather(-1, pphy2mlog)
    pphyrank = phyrank.gather(-1, pphy2phy).view(num_layers, -1)
    logcnt = mlogcnt.view(num_layers, -1).gather(-1, log2mlog)
    return pphy2log, pphyrank, logcnt


def rebalance_experts(
    weight: torch.Tensor,
    num_replicas: int,
    num_groups: int,
    num_nodes: int,
    num_gpus: int,
) -> tuple[torch.Tensor, torch.Tensor, torch.Tensor]:
    """
    Entry point for expert-parallelism load balancer.

    Parameters:
        weight: [layers, num_logical_experts], the load statistics for all
            logical experts
        num_replicas: number of physical experts, must be a multiple of
            `num_gpus`
        num_groups: number of expert groups
        num_nodes: number of server nodes, where the intra-node network
            (e.g, NVLink) is faster
        num_gpus: number of GPUs, must be a multiple of `num_nodes`

    Returns:
        physical_to_logical_map: [layers, num_replicas], the expert index of
            each replica
        logical_to_physical_map: [layers, num_logical_experts, X], the replica
            indices for each expert
        expert_count: [layers, num_logical_experts], number of physical
            replicas for each logical expert
    """
    num_layers, num_logical_experts = weight.shape
    weight = weight.float()
    if num_groups % num_nodes == 0:
        # use hierarchical load-balance policy
        phy2log, phyrank, logcnt = rebalance_experts_hierarchical(
            weight, num_replicas, num_groups, num_nodes, num_gpus)
    else:
        # use global load-balance policy
        phy2log, phyrank, logcnt = rebalance_experts_hierarchical(
            weight, num_replicas, 1, 1, num_gpus)
    num_redundant_experts = num_replicas - num_logical_experts
    maxlogcnt = num_redundant_experts + 1
    log2phy: torch.Tensor = torch.full(
        (num_layers, num_logical_experts, maxlogcnt),
        -1,
        dtype=torch.int64,
        device=logcnt.device,
    )
    log2phy.view(num_layers, -1).scatter_(
        -1,
        phy2log * maxlogcnt + phyrank,
        torch.arange(num_replicas, dtype=torch.int64,
                     device=log2phy.device).expand(num_layers, -1),
    )
    return phy2log, log2phy, logcnt
